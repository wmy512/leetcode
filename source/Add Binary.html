Given two binary strings, return their sum (also a binary string).

For example,
a = "11"
b = "1"
Return "100".


#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>


class Solution {
public:
    string addBinary( string a, string b) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        vector< int> va, vb;
        GetVectorFromString(va, a);
        GetVectorFromString(vb, b);
        int max_len = max(va.size(), vb.size());
        va.resize(max_len);
        vb.resize(max_len);  //结果写到vb
        AddVectors(va, vb);
        string result;
        GetVectorToString(vb, result);
        return result;
       
    }
    void GetVectorFromString( vector< int> & vi, string a) {
        int length = a.size();
        while (--length >= 0) {
            vi.push_back( a[length] - '0');
        }
    }
    void GetVectorToString( vector< int> & vi, string & a) {
        int len = vi.size();
        while (--len >= 0) {
            a.push_back( vi[len] + '0');
        }
    }
    void AddVectors( vector< int> & va, vector< int> & vb) {
        int carry = 0;
        for ( int i = 0; i < va.size(); i++) {
            int tmp_add = va[i] + vb[i] + carry;
            if (tmp_add >= 2) {
                tmp_add -= 2;
                carry = 1;
            }
            else {
                carry = 0;
            }
            vb[i] = tmp_add;
        }
        if (carry == 1) {
            vb.push_back(carry);
        }
    }
};

int main() {
    Solution so;
    cout << so.addBinary( "11", "1" ) << endl;
    return 0;
}
