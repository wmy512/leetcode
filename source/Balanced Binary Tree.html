Given a binary tree, determine if it is height-balanced.

For this problem, a height-balanced binary tree is defined as a binary tree in which the depth of the two subtrees of every node never differ by more than 1.

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public :
    //每次都要遍历树
    bool isBalanced(TreeNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( root == NULL) {
            return true;
        }
        int left_height = TreeHeight( root->left);
        int right_height = TreeHeight( root->right);
       
        if (abs(left_height - right_height) <= 1) {
            return isBalanced( root->left) && isBalanced( root->right);
        }
        return false;
       
    }
    int TreeHeight(TreeNode * root) {
        if ( root == NULL) {
            return 0;
        }
        int left = TreeHeight( root->left);
        int right = TreeHeight( root->right);
        return max(left + 1, right + 1);
    }
};

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public :
    bool isBalanced(TreeNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 遍历的同时完成检测
        // 由于搜索只是log(N) 数量级，优化时间不是很多，从76ms=>72ms
       
        int height;
        return IsBalancedCore( root, height);
       
    }
    bool IsBalancedCore(TreeNode * root, int & height) {
        if ( root == NULL) {
            height = 0;
            return true;
        }
        int left_height;
        int left_balance = IsBalancedCore( root->left, left_height);
        int right_height;
        int right_balance = IsBalancedCore( root->right, right_height);
        if (left_balance && right_balance && (abs(left_height - right_height) <= 1)) {
            height = max(left_height + 1, right_height + 1);
            return true;
        }
        else {
            height = max(left_height + 1, right_height + 1);
            return false;
        }
    }
};
