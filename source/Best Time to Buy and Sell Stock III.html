Say you have an array for which the ith element is the price of a given stock on day i.

Design an algorithm to find the maximum profit. You may complete at most two transactions.

Note:
You may not engage in multiple transactions at the same time (ie, you must sell the stock before you buy again).

#include <iostream>
using namespace std;
#include <vector>

class Solution {
public :
    int maxProfit( vector< int> & prices) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：前向遍历一次，找到每个位置的最大利润；反向遍历一次，找到每个位置的最小利润（都是负数）
        int length = prices.size();
        if (length < 2) {
            return 0;
        }
       
        vector< int> for_profit(length);
        vector< int> back_profit(length + 1); //to keep the zero value
       
        // forward
        int min_v = prices[0];
        int max_profit = prices[1] - prices[0];
        if (max_profit < 0) {
            max_profit = 0;
        }
        for_profit[1] = max_profit;
        for ( int i = 2; i < length; i++) {
            if ( prices[i - 1] < min_v) {
                min_v = prices[i - 1];
            }
            if (max_profit < prices[i] - min_v) {
                max_profit = prices[i] - min_v;
            }
            for_profit[i] = max_profit;
        }
       
        //backward
        int max_v = prices[length - 1]; //keep the big one
        int min_profit = prices[length - 2] - prices[length - 1];
        if (min_profit > 0) {
            min_profit = 0;
        }
        back_profit[length - 2] = min_profit;
        for ( int i = length - 3; i >= 0; i--) {
            if (max_v < prices[i + 1]) {
                max_v = prices[i + 1];
            }
            if ( prices[i] - max_v < min_profit) {
                min_profit = prices[i] - max_v;
            }
            back_profit[i] = min_profit;
        }
       
        //get the max profit
        int result = 0;
        for ( int i = 1; i < length; i++) {
            if (result < for_profit[i] - back_profit[i]) { //not i + 1
                result = for_profit[i] - back_profit[i];
            }
        }
       
        return result;
    }
};
int main() {

    int a[] = {1, 4, 3, 4, 2, 6};
    vector< int> vi(a, a + 6);
    Solution so;
    cout << so.maxProfit(vi); //7
    return 0;
}
