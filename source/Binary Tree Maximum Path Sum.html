Given a binary tree, find the maximum path sum.

The path may start and end at any node in the tree.

For example:
Given the below binary tree,

       1
      / \
     2   3

Return 6.


#include <iostream>
using namespace std;
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

//Definition for binary tree
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};
class Solution {
public:
    int maxPathSum( TreeNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        int max_sum = INT_MIN;
        MaxPathSum( root, max_sum);
        return max_sum;
    }
    int MaxPathSum( TreeNode * p, int & max_sum) {
        // 要注意避免重复计算，最大值和返回值有区别
        // 第一次出错，未考虑只有一个元素的情况
        // 第二次出错，中间是负，左右是正的考虑不对，应该是全加
        // 第三次改进，由于不管p->val是正还是负，后面的考虑都是一样的，即：
        // left >= 0 && right >= 0 max_sum是都加，sum_path加一条
        // left >= 0 && right < 0  sum_path: cur + left ，update max_sum
        // left < 0 && right >= 0  sum_path: cur + right ，update max_sum
        // left < 0 && right < 0   sum_path: cur ，update max_sum
        if ( p && p->left == NULL && p->right == NULL) {
            max_sum = max( max_sum, p->val);
            return p->val;
        }
        if ( p) {
            int left = MaxPathSum( p->left, max_sum);
            int right = MaxPathSum( p->right, max_sum);

            int left_root_right = left + p->val + right;
            int sum_path = 0;
            int cur = p->val;

            if (left >= 0 && right >= 0) {
                sum_path = max(cur + left, cur + right);
                max_sum = left_root_right > max_sum ? left_root_right : max_sum;
            }
            else if (left >= 0) { //right < 0)
                sum_path = cur + left;
                max_sum = sum_path > max_sum ? sum_path : max_sum;
            }
            else if (right >= 0) { //left < 0
                sum_path = cur + right;
                max_sum = sum_path > max_sum ? sum_path : max_sum;
            }
            else {
                sum_path = cur;
                max_sum = sum_path > max_sum ? sum_path : max_sum;
            }

            return sum_path;
        }
        return 0;
    }
};
int main() {
    TreeNode r(-1), rl(8), rr(2), rlr(-9), rrl(0), rrll(-3), rrlll(-9), rrllll(2);
    r.left = &rl;
    r.right = &rr;
    rl.right = &rlr;
    rr.left = &rrl;
    rrl.left = &rrll;
    rrll.left = &rrlll;
    rrlll.left = &rrllll;

    Solution so;
    int max_sum = 0;
    cout << so.maxPathSum(&r) << endl;
    cout << max_sum << endl;  //9
    return 0;
}
