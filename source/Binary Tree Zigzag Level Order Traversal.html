Given a binary tree, return the zigzag level order traversal of its nodes' values. (ie, from left to right, then right to left for the next level and alternate between).

For example:
Given binary tree {3,9,20,#,#,15,7},

    3
   / \
  9  20
    /  \
   15   7

return its zigzag level order traversal as:

[
  [3],
  [20,9],
  [15,7]
]

confused what "{1,#,2,3}" means? > read more on how binary tree is serialized on OJ.

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public :
    vector< vector< int> > zigzagLevelOrder( TreeNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        queue< int> level_index;
        queue< TreeNode *> level_node;
        vector< vector< int> > result;
        if ( root == NULL) {
            return result;
        }
        level_index.push(0);
        level_node.push( root);
        while (!level_node.empty()) {
            TreeNode *p = level_node.front();
            level_node.pop();
            int index = level_index.front();
            level_index.pop();

            if (index < result.size()) {
                result[index].push_back(p->val);
            }
            else {
                vector< int> vi;
                vi.push_back(p->val);
                result.push_back(vi);
            }
            if (p->left) {
                level_index.push(index + 1);
                level_node.push(p->left);
            }
            if (p->right) {
                level_index.push(index + 1);
                level_node.push(p->right);
            }
        }
        //swap the result
        for ( int i = 1 ; i < result.size(); i += 2) {
            for ( int m = 0, n = result[i].size() - 1; m < n; m++, n--) {
                swap(result[i][m], result[i][n]);
            }
        }
        return result;

    }
};
