There are N children standing in a line. Each child is assigned a rating value.

You are giving candies to these children subjected to the following requirements:

  • Each child must have at least one candy.
  • Children with a higher rating get more candies than their neighbors.

What is the minimum candies you must give?

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <map>
#include <set>


class Solution {
public:
    int candy(vector<int> &ratings) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：用set，对每一个元素cur，左遍历，直到序列不是递减为止(<=),向右同理,最后取set大的那个然后加1
        // timeout...O(n*n)
        set< int> left_set;
        set< int> right_set;
        int result = 0;
        for ( int i = 0; i < ratings.size(); i++) {
            int left_index = i - 1;
            int right_index = i + 1;
            left_set.clear();
            right_set.clear();
            while (left_index >= 0 && ratings[left_index] <= ratings[left_index + 1]) {
                if ( ratings[left_index] != ratings[i]) {
                    left_set.insert( ratings[left_index]);
                }
                left_index--;
            }
            while (right_index < ratings.size() && ratings[right_index] <= ratings[right_index - 1]) {
                if ( ratings[right_index] != ratings[i]) {
                    right_set.insert( ratings[right_index]);
                }
                right_index++;
            }
            result += max(left_set.size(), right_set.size()) + 1;
        }
        return result;
    }
    int candy2(vector<int> &ratings) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(n) 前向遍历，只要大就进行累加，并存入最大值
        int length = ratings.size();
        vector< int> result(length, 1);
        for ( int i = 1; i < length; i++) {
            if ( ratings[i - 1] < ratings[i]) {
                result[i] = max(result[i], result[i - 1] + 1);
            }
        }
        for ( int i = length - 2; i >= 0; i--) {
            if ( ratings[i + 1] < ratings[i]) {
                result[i] = max(result[i], result[i + 1] + 1);
            }
        }
        int sum = 0;
        for ( int i = 0; i < length; i++) {
            sum += result[i];
        }
        return sum;
    }

};
int main() {
