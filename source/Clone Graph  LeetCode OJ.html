Clone Graph

 Total Accepted: 1995 Total Submissions: 10968My Submissions

Clone an undirected graph. Each node in the graph contains a label and a list of its neighbors.


OJ's undirected graph serialization:

Nodes are labeled uniquely.

We use # as a separator for each node, and , as a separator for node label and each neighbor of the node.

As an example, consider the serialized graph {0,1,2#1,2#2,2}.

The graph has a total of three nodes, and therefore contains three parts as separated by #.

  1. First node is labeled as 0. Connect node 0 to both nodes 1 and 2.
  2. Second node is labeled as 1. Connect node 1 to node 2.
  3. Third node is labeled as 2. Connect node 2 to node 2 (itself), thus forming a self-cycle.

Visually, the graph looks like the following:

       1
      / \
     /   \
    0 --- 2
         / \
         \_/

#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#include <string>
#include <map>
#include <set>
#include <cmath>
#include <queue>
// Definition for undirected graph.
struct UndirectedGraphNode {
    int label;
    vector<UndirectedGraphNode *> neighbors;
    UndirectedGraphNode( int x) : label( x) {};
};
class Solution {
public:
    UndirectedGraphNode *cloneGraph( UndirectedGraphNode *node ) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 两遍bfs，第一遍找到所有的节点，然后建立map[label, UndirectedGraphNode *]的映射
        // 第二遍根据neighbors，更新复制的图
        // 300ms
        queue< UndirectedGraphNode *> bfs;
        map< int, UndirectedGraphNode *> label_nodes;
        map< int, int> label_visited;
        if ( node == NULL) {
            return node;
        }
        bfs.push( node);
        label_visited[ node->label] = 1;
        while (!bfs.empty()) {
            UndirectedGraphNode *front = bfs.front();
            bfs.pop();
            int label = front->label;
            UndirectedGraphNode *p = new UndirectedGraphNode (label);
            label_nodes[label] = p;
            //遍历neighbors
            vector< UndirectedGraphNode *> neighbors = front->neighbors;
            for ( vector< UndirectedGraphNode *>::iterator iter = neighbors.begin(); iter != neighbors.end(); ++iter) {
                if (label_visited[(*iter)->label] == 0) {
                    label_visited[(*iter)->label] = 1;
                    bfs.push((*iter));
                }
            }
        }
        label_visited.clear();
        bfs.push( node);
        label_visited[ node->label] = 1;
        while (!bfs.empty()) {
            UndirectedGraphNode *front = bfs.front();
            bfs.pop();
            UndirectedGraphNode *p = label_nodes[front->label];
            //遍历neighbors
            vector< UndirectedGraphNode *> neighbors = front->neighbors;
            for ( vector< UndirectedGraphNode *>::iterator iter = neighbors.begin(); iter != neighbors.end(); ++iter) {
                if (label_visited[(*iter)->label] == 0) {
                    label_visited[(*iter)->label] = 1;
                    bfs.push(*iter);
                }
                p->neighbors.push_back(label_nodes[(*iter)->label]);
            }
        }
        return label_nodes[ node->label];
    }
    UndirectedGraphNode *cloneGraph2( UndirectedGraphNode *node ) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 优化为一遍bfs，没有visited的map，直接使用label_nodes做判断，然后在遍历邻近表时插入
        // 264ms
        queue< UndirectedGraphNode *> bfs;
        map< int, UndirectedGraphNode *> label_nodes;
        if ( node == NULL) {
            return node;
        }
        bfs.push( node);
        while (!bfs.empty()) {
            UndirectedGraphNode *front = bfs.front();
            bfs.pop();
            int root_label = front->label;
            if (label_nodes.find(root_label) == label_nodes.end()) { //no this node
                label_nodes[root_label] = new UndirectedGraphNode(root_label);
            }
            //遍历neighbors
            vector< UndirectedGraphNode *> neighbors = front->neighbors;
            for ( vector< UndirectedGraphNode *>::iterator iter = neighbors.begin(); iter != neighbors.end(); ++iter) {
                map< int, UndirectedGraphNode *>::iterator tmp_iter = label_nodes.find((*iter)->label);
                UndirectedGraphNode *insert_node = NULL ;
                if (tmp_iter == label_nodes.end()) {
                    //如果没有找到这个节点，那么插入队列，并加入map,更新要插入的节点
                    bfs.push((*iter));
                    insert_node = new UndirectedGraphNode((*iter)->label);
                    label_nodes[(*iter)->label] = insert_node;
                }
                else {
                    //如果找到这个节点，那么更新要插入的节点
                    insert_node = label_nodes[tmp_iter->first];
                }
                label_nodes[root_label]->neighbors.push_back(insert_node);
            }
        }
        return label_nodes[ node->label];
    }

};
int main() {

    UndirectedGraphNode zero(0), one(1), two(2);
    zero.neighbors.push_back(&one);
    zero.neighbors.push_back(&two);
    one.neighbors.push_back(&two);
    two.neighbors.push_back(&two);

    UndirectedGraphNode *result = so.cloneGraph2(&zero);
    for (int i = 0; i < result->neighbors.size(); i++) {
        cout << result->neighbors[i]->label << " : ";
        UndirectedGraphNode *p = result->neighbors[i];
        for ( int j = 0; j < p->neighbors.size(); j++) {
            cout << p->neighbors[j]->label << " ";
        }
        cout << endl;
    }

}
