Combination Sum

Given a set of candidate numbers (C) and a target number (T), find all unique combinations in C where the candidate numbers sums to T.

The same repeated number may be chosen from C unlimited number of times.

Note:

  • All numbers (including target) will be positive integers.
  • Elements in a combination (a1, a2, … , ak) must be in non-descending order. (ie, a1 ≤ a2 ≤ … ≤ ak).
  • The solution set must not contain duplicate combinations.

For example, given candidate set 2,3,6,7 and target 7, 
A solution set is: 
[7] 
[2, 2, 3] 


class Solution {
public:
    vector<vector<int> > combinationSum( vector< int> & candidates, int target ) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
       
        // first sort the array, use dfs to find the solution
        sort( candidates.begin(), candidates .end());
        vector< int> one_solution;

        vector< vector< int> > result;
        dfs( candidates, 0, target , 0, one_solution, result);

        return result;
    }
    void dfs(vector<int> &candidates, int index, int target, int sum, vector<int> &one_solution, vector <vector <int > > &result) {
        if ( target == sum) {
            result.push_back( one_solution);
        }
        else if ( sum < target) {
            for ( int i = index; i < candidates.size(); i++) {
                int num = candidates[i];
                int count = target / num;

                // first verify 2 or more times of the element is satisfied.
                if (count >= 2) {
                    int sum_inner = sum;
                    one_solution.push_back(num);
                    sum_inner += num;

                    int k = 1;
                    while (k++ < count) {
                        one_solution.push_back(num);
                        sum_inner += num;
                        dfs( candidates, i + 1, target , sum_inner, one_solution, result);
                       // one_solution.pop_back();
                    }
                    k = 0;
                    while (k++ < count) {
                        one_solution.pop_back();
                    }
                }

                // second verify 1 time of the element is satisfied.
                one_solution.push_back(candidates [i]);
                dfs( candidates, i + 1, target , sum + candidates [i], one_solution , result);
                one_solution.pop_back();
 
               
            }
        }
    }
};
