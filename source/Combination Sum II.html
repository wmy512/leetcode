Given a collection of candidate numbers (C) and a target number (T), find all unique combinations in C where the candidate numbers sums to T.

Each number in C may only be used once in the combination.

Note:

  • All numbers (including target) will be positive integers.
  • Elements in a combination (a1, a2, … , ak) must be in non-descending order. (ie, a1 ≤ a2 ≤ … ≤ ak).
  • The solution set must not contain duplicate combinations.

For example, given candidate set 10,1,2,7,6,1,5 and target 8, 
A solution set is: 
[1, 7] 
[1, 2, 5] 
[2, 6] 
[1, 1, 6] 

Discuss

#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <map>


class Solution {
public:
    vector<vector<int> > combinationSum2( vector< int> & num, int target) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 加入一个map，然后统计数目，可以出现两次及以上的单独计算
        // output limit ,没有pop_back相应元素
        vector< int> one_solution;

        vector< vector< int> > result;
        map< int, int> visited;
        for ( int i = 0; i < num.size(); i++) {
            visited[ num[i]]++;
        }
        vector< int> unique_can(visited.size());
        int i = 0;
        for ( map< int, int>:: iterator iter = visited.begin(); iter != visited.end(); ++iter) {
            unique_can[i++] = (*iter).first;
        }
           
        //传入unique_candidate
        dfs(unique_can, visited, 0, target, 0, one_solution, result);

        return result;
    }
    void dfs(vector<int> &candidates, map<int, int> &visited, int index, int target, int sum , vector <int > &one_solution, vector< vector< int> > & result) {
        if ( target == sum) {
            result.push_back( one_solution);
        }
        else if ( sum < target) {
            for ( int i = index; i < candidates.size(); i++) {
                int num = candidates[i];
                int count = target / num;
                if (count >= 2 && visited[num] >= 2) {
                    // and once
                    int sum_inner = sum;
                    one_solution.push_back(num);
                    sum_inner += num;

                    // and more than one
                    int k = 1;
                    while (k++ < visited[num]) {
                        one_solution.push_back(num);
                        sum_inner += num;
                        dfs( candidates, visited , i + 1, target , sum_inner, one_solution, result );
                    }
                    k = 0;
                    while (k++ < visited[num]) {
                        one_solution.pop_back();
                    }

                }
                //only one element
                one_solution.push_back(candidates [i]);
                dfs( candidates, visited , i + 1, target , sum + candidates [i], one_solution, result );
                one_solution.pop_back();
            }
        }
    }
};
