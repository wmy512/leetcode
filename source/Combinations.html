Given two integers n and k, return all possible combinations of k numbers out of 1 ... n.

For example,
If n = 4 and k = 2, a solution is:

[
  [2,4],
  [3,4],
  [2,3],
  [1,2],
  [1,3],
  [1,4],
]


#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#include <string>

class Solution {
public:
    vector<vector<int> > combine( int n, int k) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( n < k) {
            return vector< vector< int> >();
        }
        vector< vector< int> > result;
        vector< int> tmp;
        dfs(result, tmp, n, k, 1, 0);
        return result;
    }
    void dfs(vector<vector<int> > & result, vector< int> & tmp, int n, int k, int index, int count ) {
        if ( k == count) {
            result.push_back( tmp);
            return;
        }
        for ( int i = index; i <= n; i++) {
            tmp.push_back(i);
            dfs( result, tmp, n, k, i + 1, count + 1);
            tmp.pop_back();
        }
    }
};
int main () {
    Solution so;
   
    vector<vector<int> > result = so.combine(4, 2);
    for (int i = 0; i < result.size(); i++) {
        cout << result[i][0] << "\t" << result[i][1] << endl;
    }

   
    return 0;
}
