Given inorder and postorder traversal of a tree, construct the binary tree.

Note:
You may assume that duplicates do not exist in the tree.


#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>

// Definition for binary tree
 struct TreeNode {
     int val;
     TreeNode *left;
     TreeNode *right;
     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 };

class Solution {
public:
    TreeNode *buildTree_MemoryOut( vector< int> & inorder, vector< int> & postorder) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 过多的构建临时数组，导致Memory Limit Exceeded
        if ( inorder.size() <= 0) {
            return NULL;
        }
        int element = postorder[ postorder.size() - 1];
        TreeNode *root = new TreeNode(element);
        int i = 0;
        while (i < inorder.size()) {  //find the position of this element in inorder
            if ( inorder[i] == element) {
                break;
            }
            i++;
        }
        if (i == inorder.size()) { //error
            return NULL;
        }
        vector< int> first_inorder( inorder.begin(), inorder.begin() + i);
        vector< int> second_inorder( inorder.begin() + i + 1, inorder.end());
        vector< int> first_postorder( postorder.begin(), postorder.begin() + i);
        vector< int> second_postorder( postorder.begin() + i, postorder.begin() + postorder.size() - 1);
        root->left = buildTree(first_inorder, first_postorder);
        root->right = buildTree(second_inorder, second_postorder);
        return root;
    }
    TreeNode *buildTree( vector< int> & inorder, vector< int> & postorder) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        return BuildTreeCore( inorder, 0, inorder.size(), postorder, 0, postorder.size());
    }
    TreeNode *BuildTreeCore( vector< int> & inorder, int instart, int inend, vector<int > &postorder, int poststart, int postend) {
        if ( instart >= inend) {
            return NULL;
        }
        int element = postorder[ postend - 1];
        TreeNode *root = new TreeNode(element);
        if ( postorder.size() == 1) { //last node
            return root;
        }
        int i = instart;
        while (i < inend) {  //find the position of this element in inorder
            if ( inorder[i] == element) {
                break;
            }
            i++;
        }
   
        int length = i - instart;
        root->left = BuildTreeCore( inorder, instart, instart + length, postorder, poststart, poststart + length);
        root->right = BuildTreeCore( inorder, instart + length + 1, inend, postorder, poststart + length, postend - 1);
        return root;
    }
};

int main() {
    Solution so;
    //so.solveNQueens(4);
    vector<int> in;
    in.push_back(1);
    in.push_back(3);
    in.push_back(2);

    vector<int> post;
    post.push_back(3);
    post.push_back(2);
    post.push_back(1);

    TreeNode *head = so.buildTree(in, post);
    if (head->left) {
        cout << head->left->val << endl;

    }

    return 0;
}
