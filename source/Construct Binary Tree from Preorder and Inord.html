Given preorder and inorder traversal of a tree, construct the binary tree.

Note:
You may assume that duplicates do not exist in the tree.

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public :
    TreeNode *buildTree( vector< int> & preorder, vector< int> & inorder) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        return BuildTreeCore( inorder, 0, inorder.size(), preorder, 0, preorder .size());
    }
    TreeNode *BuildTreeCore( vector< int> & inorder, int instart, int inend, vector <int > & preorder, int prestart, int preend) {
        if ( instart >= inend) {
            return NULL;
        }
        int element = preorder[ prestart];
        TreeNode *root = new TreeNode(element);
        if ( preorder.size() == 1) { //last node
            return root;
        }
        int i = instart;
        while (i < inend) {  //find the position of this element in inorder
            if ( inorder[i] == element) {
                break;
            }
            i++;
        }
   
        int length = i - instart;
        root->left = BuildTreeCore( inorder, instart, instart + length, preorder, prestart + 1, prestart + 1 + length);
        root->right = BuildTreeCore( inorder, instart + length + 1, inend, preorder , prestart + 1 + length, preend);
        return root;
    }
};
