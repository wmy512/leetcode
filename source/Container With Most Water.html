Given n non-negative integers a1, a2, ..., an, where each represents a point at coordinate (i, ai). n vertical lines are drawn such that the two endpoints of line i is at (i, ai) and (i, 0). Find two lines, which together with x-axis forms a container, such that the container contains the most water.

Note: You may not slant the container.
#include <iostream>
using namespace std;
#include <vector>
#include <stack>
#include <queue>

class Solution {
public :
    int maxArea( vector< int> & height) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(n)
        if ( height.size() <= 1) {
            return 0;
        }
        int left = 0, right = height.size() - 1;
        int max_area = 0;
        while (left < right) {
            int min_height = min( height[left], height[right]);
            max_area = max(max_area, min_height * (right - left));
            if ( height[left] < height[right]) {
                //因为left 是短板，所以不可能有 right2<right且[left, right2] 比[left, right]面积大，即最大面积已经找到，left++
                ++left;
            }
            else {
                --right;
            }
        }
        return max_area;
    }
    int maxArea2( vector< int> & height) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(N2) timeout
        if ( height.size() <= 1) {
            return 0;
        }
        int length = height.size();
        int min_index = 0;
        int max_area;
      
        for ( int i = 0; i < length; i++) {
            min_index = i;
            for ( int j = i + 1; j < length; j++) {
                if ( height[j] < height[i]) { //理解错了，只保留两边的值即可
                    min_index = j;
                }

                max_area = max(max_area, height[min_index] * (j - i));
                //cout << max_area << "," << j << "," << i << endl;
            }
        }
        return max_area;
    }
};
int main() {
    Solution so;
    int a[] = {6,14,2,11,2,7,0,9,12,7};
    vector< int> vi(a, a + 10);
  
    cout << so.maxArea2(vi) << endl;
    return 0;
}
