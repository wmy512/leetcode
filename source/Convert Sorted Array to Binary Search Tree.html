Given an array where elements are sorted in ascending order, convert it to a height balanced BST.
#include <iostream>
using namespace std;
#include <vector>

// Definition for binary tree
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode( int x) : val( x), left( NULL), right( NULL) {}
 };
class Solution {
public :
    TreeNode *sortedArrayToBST( vector< int> & num) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        int length = num.size();
        if (length <= 0) {
            return NULL;
        }
        int mid = length >> 1;
        TreeNode *p = new TreeNode( num[mid]);
       
        vector< int> left( num.begin(), num.begin() + mid);
        vector< int> right( num.begin() + mid + 1, num.end());
        p->left = sortedArrayToBST(left);
        p->right = sortedArrayToBST(right);
        return p;
    }
};

void InOrder(TreeNode *root ) {
    if ( root) {
        InOrder( root->left);
        cout << root->val << " " ;
        InOrder( root->right);
    }
}
int main() {

    int a[] = {1, 2, 3, 4, 5, 6};
    vector< int> vi(a, a + 6);
    Solution so;
    TreeNode *p = so.sortedArrayToBST(vi);
    InOrder(p);

    return 0;
}
