The count-and-say sequence is the sequence of integers beginning as follows:
1, 11, 21, 1211, 111221, ...

1 is read off as "one 1" or 11.
11 is read off as "two 1s" or 21.
21 is read off as "one 2, then one 1" or 1211.

Given an integer n, generate the nth sequence.

Note: The sequence of integers will be represented as a string.

#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>

class Solution {
public:
    string countAndSay( int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( n <= 0) {
            return "";
        }
        string prev = "1";
        for ( int i = 1; i < n; i++) {
            string next = "";
            int start = 0;
            int end = 0;
            int length = prev.size();
            while (++end < length) {
                if (prev[end] != prev[start]) {
                    char repeat_char = prev[start];
                    int repeat_time = end - start;
                    next += GetFromCharAndTimes(repeat_char, repeat_time);
                    start = end;
                }
            }
            if (start < length) {
                char repeat_char = prev[start];
                int repeat_time = length - start;
                next += GetFromCharAndTimes(repeat_char, repeat_time);
            }
            prev = next;
        }
        return prev;
    }
    string GetFromCharAndTimes( char repeat_char, int repeat_times ) {
        string result;
        char int_str[43] = {0};
        int cov_index = 0;
        while ( repeat_times) {
            int_str[cov_index++] = repeat_times % 10 + '0' ;
            repeat_times /= 10;
        }
        int tmp_len = 43;
        while (int_str[--tmp_len] == '\0') {
        }
        while (tmp_len >= 0) {
            result += int_str[tmp_len--];
        }
        result += repeat_char;
        return result;
    }
};
int main() {
    Solution so;
    cout << so.countAndSay(15) << endl;
    cout << so.GetFromCharAndTimes( '1', 23) << endl;
    return 0;
}
