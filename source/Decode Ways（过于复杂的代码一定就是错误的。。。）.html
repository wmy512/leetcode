A message containing letters from A-Z is being encoded to numbers using the following mapping:

'A' -> 1
'B' -> 2
...
'Z' -> 26

Given an encoded message containing digits, determine the total number of ways to decode it.

For example,
Given encoded message "12", it could be decoded as "AB" (1 2) or "L" (12).

The number of ways decoding "12" is 2.

#include <iostream>
using namespace std;
#include <vector>
#include <stack>
#include <string>

class Solution {
public :
    int numDecodings( string s) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 用递归很可能导致太深，最终超时。用动态规划。从最后一个字符算
        // d[i] = 1; i = s.size() - 1
        // d[i] = 0; d[i] = d[i + 1], if (s[i] 和s[i + 1] <= 26) d[i] += d[i + 2];
        // 第一次出错，没有考虑0的情况
        // 第二次出错，没有考虑10的情况
        // 第三次出错，没有考虑00的情况
        // 第四次出错，没有考虑01这种不合法的情况(首字符为0，或者连续两个字符为0，都为非法的)
        // 第五次出错，最后一位是110.。。。
        if ( s.size() == 0) {
            return 0;
        }
        vector< int> d( s.size() + 2, 1);

        for ( int i = s.size() - 1; i >= 0; i--) {
            //process one bit
            if ( s[i] == '0') { //可以处理两个0连续的情况了和首字符为0的情况
                d[i] = 0;
                continue;
            }
            else {
                d[i] = d[i + 1];
            }
            // process two bit
            string num = s.substr(i, 2);
            if (i + 1 < s.size() && num <= "26" ) {
                d[i] += d[i + 2];
            }
        }
        return d[0];
    }
   
};
int main() {
    Solution so;
    cout << so.numDecodings( "10100" ) << endl;

    return 0;
}
