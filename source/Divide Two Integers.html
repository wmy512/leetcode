Divide two integers without using multiplication, division and mod operator.

class Solution {
public:
    int divide(int dividend, int divisor) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 类似pow那种算法，有一个基数逐渐左移，直到够了一半，增加计数
        // 第一次出错，没有处理负数
        // 都转为unsigned int
        // 第二次    -2147483648, 1 超时，因为中间过程没有使用unsigned int...,而且溢出导致循环无法退出
        unsigned int udividend, udivisor;
        int neg_dividend, neg_divisor;
        if ( dividend < 0) {
            udividend = ~( dividend - 1);
            neg_dividend = -1;
        }
        else {
            udividend = dividend;
            neg_dividend = 1;
        }
        if ( divisor < 0) {
            udivisor = ~( divisor - 1);
            neg_divisor = -1;
        }
        else {
            udivisor = divisor;
            neg_divisor = 1;
        }
        int count = 0;
        while (udividend >= udivisor) {
            unsigned long long one_count = 1;
            unsigned long long one_num = udivisor;
            while ((one_num << 1) < udividend) {  //只能使用小于，否则2147483648没有办法处理
                one_count <<= 1;
                one_num <<= 1;
            }
            count += one_count;
            udividend -= one_num;
        }
        return count * neg_dividend * neg_divisor;
    }
};
