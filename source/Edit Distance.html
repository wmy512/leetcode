You have the following 3 operations permitted on a word:

a) Insert a character
b) Delete a character
c) Replace a character

/**
Given two words word1 and word2, find the minimum number of steps required to convert word1 to word2. (each operation is counted as 1 step.)

You have the following 3 operations permitted on a word:

a) Insert a character
b) Delete a character
c) Replace a character
*/

class Solution {
public:
    int minDistance( string word1, string word2) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        //min_dist = min(matrix[i - 1][j] + 1, matrix[i][j - 1] + 1, matrix[i - 1][j - 1] + (word1[i] != word2[j]));
        if ( word1.size() == 0) {  //bug one, can't be empty..
            return word2.size();
        }
        if ( word2.size() == 0) {
            return word1.size();
        }

        vector< vector< int> > matrix( word1.size() + 1, vector< int>( word2.size() + 1));

        for ( int i = 0; i < matrix.size(); i++) {
            matrix[i][0] = i;
        }
        for ( int i = 0; i < matrix[0].size(); i++) {
            matrix[0][i] = i;
        }
        for ( int i = 1; i <= word1.size(); i++) {
            for ( int j = 1; j <= word2.size(); j++) {
                int min_dist = min(matrix[i - 1][j] + 1, matrix[i][j - 1] + 1); //delete //insert
                min_dist = min(min_dist, matrix[i - 1][j - 1] + ( word1[i - 1] != word2[j - 1])); //
                matrix[i][j] = min_dist;
            }
        }
        //PrintVector(matrix);
        return matrix[ word1.size()][ word2.size()];
    }

    }

};
