There are N gas stations along a circular route, where the amount of gas at station i is gas[i].

You have a car with an unlimited gas tank and it costs cost[i] of gas to travel from station i to its next station (i+1). You begin the journey with an empty tank at one of the gas stations.

Return the starting gas station's index if you can travel around the circuit once, otherwise return -1.

Note:
The solution is guaranteed to be unique.
// 代价为0，如果总代价<0，肯定无解。代价>=0，那么遍历并累加代价，如果代价<0，说明不是本站和前面的某一个代价大于0的站开始，必将从后面的站开始（下一站）
        // 因为必有解，所以O(n)
class Solution {
public :
    int canCompleteCircuit( vector< int> & gas, vector< int> & cost) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        int sum = 0;
        int total = 0;
        int index = -1;
        for ( int i = 0; i < gas.size(); i++) {
            int diff = gas[i] - cost[i];
            sum += diff;
            total += diff;
            if (sum < 0) {
                index = i;
                sum = 0;
            }
        }
        return total >= 0 ? index + 1 : -1;
    }
};

class Solution {
public :
    int Core( vector< int> & gas, vector< int> & cost, int index) {

        int remain = 0;
               for ( int i = index; i < gas.size(); i++) {
                     remain += gas[i] - cost[i];
                      if (remain < 0) {
                            return -1;
                     }
                     
              }
               for ( int i = 0; i < index; i++) {
                     remain += gas[i] - cost[i];
                      if (remain < 0) {
                            return -1;
                     }
              }
               return index;
    }
    int canCompleteCircuit( vector< int> & gas, vector< int> & cost) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        //1. 每一站的代价为gas-cost, 也就是求从哪一站开始累加代价和总是大于0。比O(N2）好点，没有超时。。600ms
        long long gas_sum = 0, cost_sum = 0;
        for ( int i = 0; i < gas.size(); i++) {
            gas_sum += gas[i];
            cost_sum += cost[i];
        }
        //can't be a solution
        if (gas_sum < cost_sum) {
            return -1;
        }
        for ( int i = 0; i < gas.size(); i++) {
            if ( gas[i] >= cost[i]) {
                int tmp_index = Core( gas, cost, i);
                if (tmp_index != -1) {
                    return tmp_index;
                }
            }
        }
        //no solution
        return -1;
    }
