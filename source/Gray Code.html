The gray code is a binary numeral system where two successive values differ in only one bit.

Given a non-negative integer n representing the total number of bits in the code, print the sequence of gray code. A gray code sequence must begin with 0.

For example, given n = 2, return [0,1,3,2]. Its gray code sequence is:

00 - 0
01 - 1
11 - 3
10 - 2

Note:
For a given n, a gray code sequence is not uniquely defined.

For example, [0,2,3,1] is also a valid gray code sequence according to the above definition.

 *
 * 可得出基本思路，最外围中间分开，上面和下面（除最高位）对称，下面都是最高位为1，上面都是最高位为0,
 */
class Solution {
public :
    vector< int> grayCode( int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        vector< int> result;
        if ( n < 0) {
            return result;
        }
        // when n is 0, the result is 0
        int num = (1 << n);
        result.assign(pre_result.begin(), pre_result.begin() + num);
        return result;
    }
    Solution() {
        pre_result.resize(1 << 20); //the max 4MB, in the test, the max n is 11
        pre_result[0] = 0;
        //pre_result[1] = 1;
        for ( int i = 1; i <= 20; i++) {
            int total = 1 << i;
            int first_index = 0, first_max = total >> 1;
            while (first_index < first_max) {
                pre_result[total - 1 - first_index] = first_max + pre_result[first_index];
                first_index++;
            }
        }
    }
private :
    vector< int> pre_result;
};
