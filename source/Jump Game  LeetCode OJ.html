Jump Game

Total Accepted: 1719 Total Submissions: 6493 My Submissions

Given an array of non-negative integers, you are initially positioned at the first index of the array.

Each element in the array represents your maximum jump length at that position.

Determine if you are able to reach the last index.

For example:
A = [2,3,1,1,4], return true.

A = [3,2,1,0,4], return false.

Discuss

class Solution {
public:
    bool canJump(int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( A == NULL || n <= 1) {
            return true;
        }
        int cur = 0;
        for ( int i = 0; i < n; i++) {
            if ( A[i] == 0 && i >= cur) { //已经走到最远了,不能继续前进了
                return false;
            }
            cur = max(cur, A[i] + i);
            if (cur >= n - 1) { //到达终点
                return true;
            }
        }
        return true;
    }
};

class Solution {
public:
    bool canJump(int A[], int n) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        if ( A == NULL || n <= 0) {
            return false;
        }
        int first_max = 0;
        int second_max = 0;
        //每次更新first_max,和second_max两个区间段
        for ( int i = 0; i < n; ) {
            if (i > second_max) { //this position can't be reached..
                return false;
            }
            /*if (A[i] == 0 && i == second_max && second_max < n - 1) {
                return false;
            }*/
            //cout << i << endl;
            first_max = A[i] + i;
            int second = i + 1;
            second_max = max(first_max, second_max);
            int tmp_max = second_max;
            while (second <= second_max) {
                tmp_max = max(tmp_max, A[second] + second);
                second++;
            }
            second_max = tmp_max;
            if (second_max >= n - 1) {
                return true;
            }
            i = first_max + 1;
        }

        return false;
    }
};
