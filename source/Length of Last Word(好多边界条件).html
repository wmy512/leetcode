Given a string s consists of upper/lower-case alphabets and empty space characters ' ', return the length of last word in the string.

If the last word does not exist, return 0.

Note: A word is defined as a character sequence consists of non-space characters only.

For example, 
Given s = "Hello World",
return 5.

#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>

class Solution {
public:
    int lengthOfLastWord( const char * s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 第一次出错因为没有考虑 ""
        // 第二次出错因为结尾的空格（这个直接不要的） "a "
        if ( s == NULL) {
            return 0;
        }
        int length = 0;
        while ( s[length]) {
            length++;
        }
        // trim the tail blank
        while ( s[--length] == ' ') {
        }

        int space_index = -1;
        int i = 0;
        while (i <= length) {
            if ( s[i] == ' ') {
                space_index = i;
            }
            i++;
        }
        if (space_index == -1) { //bug one: no blank
            return i;
        }
        return i - space_index - 1;
    }
};
int main() {
    Solution so;

    cout << so.lengthOfLastWord( "hello world") << endl; //1 bug one
    return 0;
}
