Given a digit string, return all possible letter combinations that the number could represent.

A mapping of digit to letters (just like on the telephone buttons) is given below.

Input:Digit string "23"
Output: ["ad", "ae", "af", "bd", "be", "bf", "cd", "ce", "cf"].

Note:
Although the above answer is in lexicographical order, your answer could be in any order you want.

#include <iostream>
using namespace std;
#include <vector>
#include <string>


class Solution {
public:
    vector<string> letterCombinations( string digits) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        const char *mapdigit[] = { "", "" , "abc" , "def" , "ghi" , "jkl" , "mno" , "pqrs", "tuv" , "wxyz" };
        vector< string> result;
        if ( digits.size() <= 0) {
            //第一次出错，默认返回[""]
            result.push_back( "");
            return result;
        }
        string str;
        dfs(result, str, digits, mapdigit, 0);
        return result;
    }
    void dfs(vector<string> & result, string & str, string digits, const char **mapdigit, int index ) {
        if ( index == digits.size()) {
            result.push_back( str);
            return;
        }
        int map_index = digits[ index] - '0';
        for ( int i = 0; mapdigit[map_index][i] != '\0'; i++) {
            str += mapdigit[map_index][i];
            dfs( result, str, digits, mapdigit, index + 1);
            str.erase( str.begin() + str.size() - 1);
        }
        // 避免123无法执行的情况
        if ( mapdigit[map_index][0] == '\0') {
            dfs( result, str, digits, mapdigit, index + 1);
        }
    }

};
int main() {
   
    Solution so;
    vector<string> result = so.letterCombinations( "1203");
    for (int i = 0; i < result.size(); i++) {
        cout << result[i] << " ";
    }

    return 0;
}
