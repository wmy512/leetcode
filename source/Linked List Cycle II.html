Given a linked list, return the node where the cycle begins. If there is no cycle, return null.

Follow up:
Can you solve it without using extra space?

Discuss


/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle( ListNode * head) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        //先判断是否有环，没有直接返回NULL
        ListNode *fast = head;
        ListNode *slow = head;
        bool has_circle = false;
       
        while (fast && slow) {
            fast = fast->next;
            if (fast) {
                fast = fast->next;
            }
            slow = slow->next;
            if (fast != NULL && slow != NULL && slow == fast) {
                has_circle = true;
                break;
            }
        }
        if (has_circle) {
            // 断开节点，然后找到第一个相同的点
            ListNode *p1 = head, *p2 = slow->next;
            slow->next = NULL;
            //找到两个的各自的长度，因为如果有同一个交点，那么从那个点开始到最后都是一样的
            int p1_length = 0, p2_length = 0;
            ListNode *p = p1;
            while (p) {
                p1_length++;
                p = p->next;
            }
            p = p2;
            while (p) {
                p2_length++;
                p = p->next;
            }
            //update the longer list
            int min_diff = 0;
            if (p1_length < p2_length) {  //p2 is longer
                min_diff = p2_length - p1_length;
                int i = 0;
                while (i < min_diff) {
                    p2 = p2->next;
                    i++;
                }
            }
            else { //p1 is longer
                min_diff = p1_length - p2_length;
                int i = 0;
                while (i < min_diff) {
                    p1 = p1->next;
                    i++;
                }
            }
            int max_length = max(p1_length, p2_length);
            while (min_diff < max_length) {
                if (p1 == p2) {
                    return p1;
                }
                p1 = p1->next;
                p2 = p2->next;
                min_diff++;
            }
        }
        else {
            return NULL;
        }
    }
};


/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle( ListNode * head) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        if ( head == NULL) {
            return head;
        }
        ListNode *slow = head;
        ListNode *fast = head;
        bool is_circle = false;
        //先判断有没有环
        //思路都是根据有环情况下做出的假设
        //当slow首先进入环的时候，走了k步，那么fast走了2k步，在环内走了k步，两者相差LOOP_SIZE-k步，fast比slow速度快，平均一次缩短1的距离。
        //经过LOOP_SIZE-k次循环，即slow走了LOOP_SIZE-k步，距离入口(LOOP_SIZE-(LOOP_SIZE-k))=k步，而head距离入口也是k步，所以让slow回归head，
        //fast不动，两者同样速度移动，最终相遇点就是入口。。经典！！！
        while (slow && fast) {
            fast = fast->next;
            if (fast) {
                fast = fast->next;
            }
            slow = slow->next;
            if (slow && fast && slow == fast) {
                is_circle = true;
                break;
            }
        }
        if (is_circle) {
            slow = head;
            while (slow != fast) {
                slow = slow->next;
                fast = fast->next;
            }
            return slow;
        }
        else {
            return NULL;
        }
    }
};
