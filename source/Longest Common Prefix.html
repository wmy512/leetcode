Write a function to find the longest common prefix string amongst an array of strings.
class Solution {
public :
    string longestCommonPrefix( vector< string> & strs) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( strs.size() == 0) {
            return "" ;
        }
        // find the minimum length string. 提高速度 32ms->24ms
        string first = strs[0];
       
        int min_index = 0;
        int min_length = first.size();
        for ( int i = 1; i < strs.size(); i++) {
            if ( strs[i].size() < min_length) {
                min_length = strs[i].size();
                min_index = i;
            }
        }
        first = strs[min_index];
       
        string result = "" ;
        for ( int c_index = 0; c_index < first.size(); c_index++) {
            for ( int i = 0; i < strs.size(); i++) {
                if (c_index >= strs[i].size() || strs[i][c_index] != first[c_index]) {
                    return result;
                }
            }
            result += first[c_index];
        }
        return result;
    }
};
