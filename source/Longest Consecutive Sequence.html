Given an unsorted array of integers, find the length of the longest consecutive elements sequence.

For example,
Given [100, 4, 200, 1, 3, 2],
The longest consecutive elements sequence is [1, 2, 3, 4]. Return its length: 4.

Your algorithm should run in O(n) complexity.


#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>
#include <unordered_set>

class Solution {
public:
    int longestConsecutive( vector< int> & num) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(n * log(n))  68 ms
        sort( num.begin(), num.end());
        int count = 1;
        int max_count = 1;
        for ( int i = 1; i < num.size(); i++) {
            if ( num[i] - 1 == num[i - 1]) {
                count++;
                if (max_count < count) {
                    max_count = count;
                }
            }
            else if ( num[i] == num[i - 1]) {  // equal ignore
                continue;
            }
            else if ( num[i] - 1 != num[i - 1]) {
                count = 1;
            }
        }
        return max_count;
    }

    int longestConsecutive_Speed( vector< int> & num) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(n)   80 ms....
        unordered_set<int > flags;
        for ( int i = 0; i < num.size(); i++) {
            flags.insert( num[i]);
        }
        int max_count = 0;

        for ( int i = 0; i < num.size(); i++) {
            int cur = LongestConsecutiveSpeedCore(flags, num[i], BACKWARD);
            int after = LongestConsecutiveSpeedCore(flags, num[i] + 1, FORWARD);
            if (max_count < cur + after) {
                max_count = cur + after;
            }
        }
        return max_count;
    }
   
   
private:
    enum DIRECTION {
        BACKWARD = -1,
        FORWARD = 1,
    };
    int LongestConsecutiveSpeedCore( unordered_set<int > &flags, int value, DIRECTION dir ) {
        int count = 0;
        while ( flags.find( value) != flags.end()) {
            count++;
            flags.erase( value);
            value += dir;
        }
        return count;
    }

};
