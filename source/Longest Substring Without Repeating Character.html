Given a string, find the length of the longest substring without repeating characters. For example, the longest substring without repeating letters for "abcabcbb" is "abc", which the length is 3. For "bbbbb" the longest substring is "b", with the length of 1.

Discuss

#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <map>
#include <set>

class Solution {
public:
    int lengthOfLongestSubstring( string s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(2n) 64ms
        // 类似于min_window
        int cur[256] = {0};
        int max_len = 0;
        int j = 0;
        for ( int i = 0; i < s.size(); i++) {
            cur[ s[i]]++;
            if (cur[ s[i]] == 2) { //include repeat
                int max_v = i - j;
                max_len = max(max_len, max_v);
                while (j <= i && cur[ s[j]] <= 1) {
                    cur[ s[j++]]--;
                }
                cur[ s[j++]]--; //跨过这一个
            }
        }
        max_len = max(( int) s.size() - j, max_len); //处理最后一个
        return max_len;
    }
};
class Solution {
public:
    int lengthOfLongestSubstring( string s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // O(26n)
        // 第一次出错因为没有更新visited[s[j]]
        int visited[256] = {0};
        int max_len = 0;
        for ( int i = 0; i < s.size(); i++) {
            memset(visited, 0, sizeof(visited));
            visited[ s[i]] = 1;
            int j = i + 1;
            for (; j < s.size(); j++) {
                if (visited[ s[j]]) {
                    break;
                }
                visited[ s[j]] = 1; // update
            }
            int length = j - i;
            max_len = max(max_len, length);
        }
        return max_len;
    }
};
