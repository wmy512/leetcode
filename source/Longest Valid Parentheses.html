Given a string containing just the characters '(' and ')', find the length of the longest valid (well-formed) parentheses substring.

For "(()", the longest valid parentheses substring is "()", which has length = 2.

Another example is ")()())", where the longest valid parentheses substring is "()()", which has length = 4.

Discuss

class Solution {
public:
    int longestValidParentheses( string s) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 一维动态规划求解 dp[i]表示s[i - 1]..s[length - 1] 的匹配长度，
        // dp[0..s.length - 1] = 0;
        // 从后向前算，如果s[i] == '(',那么查看 j = i + dp[i + 1] + 1的位置是否是')'
        // 如果是')'那么dp[i] += dp[i + 1] + 2; 并加上dp[j + 1]
        vector< int> dp( s.size() + 2); // 避免溢出
        int max_len = 0;
        for ( int i = s.size() - 2; i >= 0; i--) {
            if ( s[i] == '(') {
                int j = i + dp[i + 1] + 1;
                if (j < s.size() && s[j] == ')') {
                    dp[i] = dp[i + 1] + 2;
                    dp[i] += dp[j + 1];
                }
                max_len = max(max_len, dp[i]);
            }
        }
        return max_len;
    }
};

class Solution {
public:
    int longestValidParentheses( string s) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 针对每个"("遍历，遇到"("加1，遇到")"减一，一旦<0break，并更新最大程度，最后也要更新一次
        // 加入一个优化措施，比如如果剩余字符串不到max_len直接退出
        // 仍然超时。。。。

        int max_len = 0;
        for ( int i = 0; i < s.size(); i++) {
            if ( s[i] == ')') {
                continue;
            }
            int tmp_len = 0;
            int match = 0;

            for ( int j = i; j + max_len < s.size(); j++) {
                if ( s[j] == '(') {
                    match++;
                }
                else if ( s[j] == ')') {
                    match--;
                    if (match == 0) {
                        tmp_len = j - i + 1;
                    }
                    else if (match < 0) {
                        break;
                    }
                }
            }
            max_len = max(max_len, tmp_len);
        }
        return max_len;
    }
};
