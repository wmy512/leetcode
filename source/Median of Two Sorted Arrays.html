There are two sorted arrays A and B of size m and n respectively. Find the median of the two sorted arrays. The overall run time complexity should be O(log (m+n)).
而对于“Kth element in 2 sorted array”， 如下图，两个中位数 A[m/2] 和 B[n/2]， 可以将数组划分为四个部分。而丢弃哪一个部分取决于两个条件：1， (m/2 + n/2)?k；2，A[m/2] ? B[n/2];


如果 (m/2 + n/2) > k，那么意味着，当前中位数取高了，正确的中位数要么在 Section 1或者Section3中。如果A[m/2] > B[n/2], 意味着中位数肯定不可能在Section 2里面，那么新的搜索可以丢弃这个区间段了。同理可以推断出余下三种情况，如下所示：

If (m/2+n/2+1) > k && am/2 > bn/2 , drop Section 2
If (m/2+n/2+1) > k && am/2 < bn/2 , drop Section 4
If (m/2+n/2+1) < k && am/2 > bn/2 ,  drop Section 3
If (m/2+n/2+1) < k && am/2 < bn/2 ,  drop Section 1


简单的说，就是或者丢弃最大中位数的右区间，或者丢弃最小中位数的左区间。

#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>
#include <unordered_set>
 
class Solution {
public:
    double findMedianSortedArrays( int A[], int m, int B[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.

        // 第一次出错因为 错误判断了m，n，不能处理 [],[1]
        int length = m + n;
        if (length & 1 == 1) {
            return findKth( A, m, B, n, length / 2 + 1);
        }
        return (findKth( A, m, B, n, length / 2) + findKth( A, m, B, n, length / 2 + 1)) / 2.0;
       
    }
    double findKth( int A[], int m, int B[], int n, int k) {
        //O(n)
        int count = 0;
        int im = 0, in = 0;
        while (im < m && in < n) {
            int element = A[im];
            if ( B[in] < A[im]) {
                element = B[in];
                in++;
            }
            else {
                element = A[im];
                im++;
            }
            if (++count == k) {
                return element;
            }
        }
        if (im < m) {
            return A[im + k - count - 1];
        }
        if (in < n) {
            return B[in + k - count - 1];
        }
    }
    double findKth( int A[], int m, int B[], int n, int k) {
        if (m > n) {
            return findKth(B, n, A, m, k); //for simplicy, m is small one
        }
        if (m <= 0) {
            return B[k - 1];
        }
        if (k == 1) {
            return min(A[0], B[0]);
        }
        int ia = min(k / 2, m);
        int ib = k - ia;
        if (A[ia - 1] < B[ib - 1]) {
            return findKth(A + ia, m - ia, B, n, k - ia); //A的左边不要
        }
        else if (A[ia - 1] > B[ib - 1]) {
            return findKth(A, m, B + ib, n - ib, k - ib); //B的左边不要
        }
        else {
            return A[ia - 1];
        }
    }
    double findKth( int A[], int m, int B[], int n, int k) {
     
        //递归的终止条件如n==0, m ==0及k==1的处理
        if (m <= 0) {
            return B[k - 1];
        }
        if (n <= 0) {
            return A[k - 1];
        }
        if (k == 1) {
            return min(A[0], B[0]);
        }
        int m_mid = m / 2;
        int n_mid = n / 2;
        if (A[m_mid] < B[n_mid]) {
            if (m_mid + n_mid + 1 < k) {
                return findKth(A + m_mid + 1, m - (m_mid + 1), B, n, k - (m_mid + 1)); //A的左边不要(正好不要m_mid)
            }
            else { // k <= m_mid + n_mid + 1
                return findKth(A, m, B, n_mid, k); //B的右边不要(正好不要n_mid)
            }
        }
        else {
            if (m_mid + n_mid + 1 < k) {
                return findKth(A, m, B + n_mid + 1, n - (n_mid + 1), k - (n_mid + 1)); //B的左边不要(正好不要n_mid)
            }
            else {
                return findKth(A, m_mid, B, n, k); //A的右边不要(正好不要m_mid)
            }
        }
    }
};
int main() {
    Solution so;
   
    int A[] = {2};
    int B[] = {3};
    cout << so.findMedianSortedArrays(A, 1, B, 1) << endl;
   
    return 0;
}
