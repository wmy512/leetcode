Merge Two Sorted Lists

Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *mergeTwoLists(ListNode * l1, ListNode * l2) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( l1 == NULL && l2 == NULL) {
            return NULL;
        }
        if ( l1 == NULL) {
            return l2;
        }
        if ( l2 == NULL) {
            return l1;
        }
        ListNode *cur1 = l1, *cur2 = l2;
        ListNode *result_head = NULL, *result_cur = NULL;
       
        while (cur1 && cur2) {
            if (cur1->val < cur2->val) {
                if (result_head == NULL) {
                    result_head = cur1;
                }
                else {
                    result_cur->next = cur1;
                }
                result_cur = cur1;
                cur1 = cur1->next;
            }
            else {
                if (result_head == NULL) {
                    result_head = cur2;
                }
                else {
                    result_cur->next = cur2;
                }
                result_cur = cur2;
                cur2 = cur2->next;
            }
        }
        result_cur->next = NULL; //in case both cur1 and cur2 is null
        if (cur1) {
            result_cur->next = cur1;
        }
        if (cur2) {
            result_cur->next = cur2;
        }
        return result_head;
    }
};
