Merge k sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

#include <iostream>
using namespace std;
#include <vector>
#include <string>

#include <queue>

// Definition for singly-linked list.
  struct ListNode {
     int val;
     ListNode *next;
     ListNode( int x) : val( x), next( NULL) {}
 };

class Solution {
private :
    struct CompareListNode {
        bool operator()( const ListNode * lhs, const ListNode * rhs) {
            return lhs->val > rhs->val;
        }
    };
public :
    ListNode *mergeKLists( vector< ListNode *> & lists) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 复杂度 O(klog(k) + (N - k)log(k))  N是所有元素个数
        // 第一次出错，[{}] 无法处理,添加进入优先队列时进行判断
        priority_queue <ListNode *, vector <ListNode *>, CompareListNode > pq;
        ListNode *result_head = NULL;
        ListNode *result_tail = NULL;

        for ( int i = 0; i < lists.size(); i++) {
            //push the first element
            ListNode *head = lists[i];
            if (head != NULL) {
                pq.push(head);
            }
        }
        while (!pq.empty()) {
            ListNode *front = pq.top();
            pq.pop();
            //push the next element
            if (front->next != NULL) {
                pq.push(front->next);
            }
            //build the new list
            if (result_head == NULL) {
                result_head = front;
            }
            else {
                result_tail->next = front;
            }
            result_tail = front;
        }
        return result_head;
    }
};
