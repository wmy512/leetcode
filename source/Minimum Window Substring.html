Given a string S and a string T, find the minimum window in S which will contain all the characters in T in complexity O(n).

For example,
S = "ADOBECODEBANC"
T = "ABC"

Minimum window is "BANC".

Note:
If there is no such window in S that covers all characters in T, return the emtpy string "".

If there are multiple such windows, you are guaranteed that there will always be only one unique minimum window in S.

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <string>
#include <queue>

class Solution {
public:
    string minWindow( string S, string T) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // map1 => 对应T中出现的字符
        // map2 => 对应是否出现，如果重复出现更新下标索引 O(256*n)
        // 第一次出错，理解错了题意。。重复字符需要考虑
        // timeout
        int map1[256] = {0};
        int map2[256] = {0};
        vector< queue< int> > visited(256);
       
       
        for ( int i = 0; i < T.size(); i++) {
            map1[ T[i]]++;
            map2[ T[i]]++;
        }
       
        int count = 0;
        string result = "";
        for ( int i = 0; i < S.size(); i++) {
            if (map1[ S[i]]) {
               
                //遍历map2看是否为都为0
                visited[ S[i]].push(i);
                map2[ S[i]]--;
                if (map2[ S[i]] < 0) {
                    map2[ S[i]] = 0;
                    visited[ S[i]].pop(); //说明那个字符多出现一个，进行计数的更新，并将下标移出队列
                }
                if (map2[ S[i]] > 0) {
                    continue;
                }
                bool zero_flag = true;
                for ( int j = 0; j < 256; j++) {
                    if (map2[j] != 0) {
                        zero_flag = false;
                        break;
                    }
                }
                if (zero_flag) { //全为空，找到最小下标
                    //找到最小的下标索引，然后与当前下标一同获得window
                    int min_index = INT_MAX;
                    for ( int j = 0; j < 256; j++) {
                        if (visited[j].size() > 0) {
                            min_index = min(min_index, visited[j].front()); //只取最前面的元素下标
                        }
                    }
                    if (result == "" || i - min_index + 1 < result.size()) {
                        result = S.substr(min_index, i - min_index + 1);
                    }
                }
            }
        }
        return result;
    }
    string minWindow2( string S, string T) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        //上面那个还是太慢了，虽然行的通。。
        //下面是两次遍历，贪心算法
        // 56ms
        int map1[256] = {0}; // 对应T中出现的字符及数量
        int map2[256] = {0};

        for ( int i = 0; i < T.size(); i++) {
            map1[ T[i]]++;
        }
        int left = 0, right = 0, min_len = INT_MAX;
        //minn和left，right分别记录最小子串的最小值和左右位置。
        int count = 0; //记录是否满足条件
        for ( int i = 0, j = 0; i < S.size(); i++) {
            if (map2[ S[i]] < map1[ S[i]]) { //如果S中不存在T中字符，那么这个不会成立
                count++;
            }
            map2[ S[i]]++;
            if (count == T.size()) {
                while (j <= i && map2[ S[j]] > map1[ S[j]]) { //直到j 到达最小限制，再前进，就不能满足条件了,只要S中的字符T中不存在，一定会继续更新j
                    --map2[ S[j]];
                    ++j;
                }
                int tmp_len = i - j + 1;
                if (tmp_len < min_len) {
                    left = j;
                    right = i;
                    min_len = i - j + 1;
                }
                if (j <= i) {  //下一次如果还有满足条件的window，那么当前索引肯定要变了
                    --map2[ S[j]];
                    count--;
                    j++;
                }
            }
        }
        if (min_len < INT_MAX) {
            return S.substr(left, min_len);
        }
        return "";

    }
};
