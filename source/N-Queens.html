The n-queens puzzle is the problem of placing n queens on an n×n chessboard such that no two queens attack each other.

Given an integer n, return all distinct solutions to the n-queens puzzle.

Each solution contains a distinct board configuration of the n-queens' placement, where 'Q' and '.' both indicate a queen and an empty space respectively.

For example,
There exist two distinct solutions to the 4-queens puzzle:

[
 [".Q..",  // Solution 1
  "...Q",
  "Q...",
  "..Q."],

 ["..Q.",  // Solution 2
  "Q...",
  "...Q",
  ".Q.."]
]



#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>

class Solution {
public:
    vector<vector<string> > solveNQueens( int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        vector< int> columns( n);
        vector< int> slash_l(2 * n - 1);
        vector< int> slash_r(2 * n - 1);
       
        int count = 0;
        vector< vector< string> > result;
        vector< string> one;
        dfs(columns, slash_l, slash_r, 0, n, count, result, one);
        return result;

    }
    void dfs(vector<int> &columns, vector<int> &slash_l, vector<int> &slash_r,
        int index, int n, int & count, vector< vector< string> > & result, vector<string > &one) {
        if ( index == n) {
            count++;
            result.push_back( one);
            return;
        }
        for ( int i = 0; i < n; i++) { //每行填一个
            string str( n, '.');
           
            if ( columns[i] || slash_l[ n - index - 1 + i] || slash_r[ index + i]) {
                continue;  //有冲突
            }
            str[i] = 'Q';
            columns[i] = slash_l[ n - index - 1 + i] = slash_r[ index + i] = 1;
            one.push_back(str);
            dfs( columns, slash_l, slash_r, index + 1, n, count, result, one);
            //restore
            columns[i] = slash_l[ n - index - 1 + i] = slash_r[ index + i] = 0;
            one.pop_back();
            str[i] = '.';
        }

    }
};


int main() {
    Solution so;
    so.solveNQueens(4);
    return 0;
}
