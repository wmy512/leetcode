Given a string s, partition s such that every substring of the partition is a palindrome.

Return the minimum cuts needed for a palindrome partitioning of s.

For example, given s = "aab",

Return 1 since the palindrome partitioning ["aa","b"] could be produced using 1 cut

#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <string>


class Solution {
public:
    bool IsPalindrome( string str) {
        for ( int start = 0, end = str.size() - 1; start < end; start++, end--) {
            if ( str[start] != str[end]) {
                return false;
            }
        }
        return true;
    }
    int minCut(string s) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        map< string, int> visited;
        int min_cnt = FindMinCut( s, 0, 0, visited);
        return min_cnt;
    }
    int FindMinCut( string & s, int index, int cur_cnt, map< string, int> & visited) {
        if ( index >= s.size()) {
            cur_cnt--;
            return cur_cnt;
        }
        int part_min = INT_MAX;

        //改成按最大长度测试，这样后面的优化判断才会起作用
        for ( int i = s.size(); i >= index + 1; i--) {
        //for (int i = index + 1; i <= s.size(); i++) {
            string part = s.substr( index, i - index);
            if (IsPalindrome(part)) {
                string rest = s.substr(i);

                cur_cnt++;
                //优化搜索，如果剩余字符串已经找到过最小分割，那么直接加上当前的就好了
                if (rest != "") {
                    map< string, int>:: iterator iter = visited.find(rest);
                    if (iter != visited.end()) {
                        part_min = min(part_min, cur_cnt + iter->second);
                        cur_cnt--;
                        continue;
                    }
                }
                // 单词为空，或者在visited里面没有找到
                int part_cnt = FindMinCut( s, i, cur_cnt, visited);
                if (rest != "") {
                    visited[rest] = part_cnt - cur_cnt; //保存字符串的最小分割
                }
                part_min = min(part_min, part_cnt);

                cur_cnt--;
            }
            //优化判断
            int residual = 0;
            if (i == s.size() - 1) {
                residual = cur_cnt;
            }
            else {
                residual = cur_cnt + 1;
            }
            if (residual >= part_min) {
                break;
            }
        }

        return part_min;
    }
};

    return 0;
}
