Given a string s, partition s such that every substring of the partition is a palindrome.

Return all possible palindrome partitioning of s.

For example, given s = "aab",
Return

  [
    ["aa","b"],
    ["a","a","b"]
  ]

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <map>
#include <set>

class Solution {
public:
    vector<vector<string>> partition( string s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        vector< vector< string> > result;
        vector< string> one;
        dfs(result, one, s, 0);
        return result;
    }
    bool IsPalindrome( string str) {
        for ( int start = 0, end = str.size() - 1; start < end; start++, end--) {
            if ( str[start] != str[end]) {
                return false;
            }
        }
        return true;
    }
    void dfs(vector<vector<string> > & result, vector< string> & one, string & s, int index) {
        if ( index == s.size()) {
            result.push_back( one);
            return;
        }
        for ( int i = index + 1; i <= s.size(); i++) {
            string sub_str = s.substr( index, i - index);
           
            //cout << "sub_str = " << sub_str << endl;
            if (IsPalindrome(sub_str)) {
                one.push_back(sub_str);
                dfs( result, one, s, i);
                one.pop_back();
            }
        }
    }

};
int main() {
   
    Solution so;
    vector<vector<string> > result = so.partition( "");
    for (int i = 0; i < result.size(); i++) {
        for ( int j = 0; j < result[i].size(); j++) {
            cout << result[i][j] << " ";
        }
        cout << endl;
    }
   
    return 0;
}
