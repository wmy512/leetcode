Given a number represented as an array of digits, plus one to the number.

#include <iostream>
using namespace std;
#include <vector>
#include <stack>
#include <queue>

class Solution {
public :
    vector< int> plusOne( vector< int> & digits) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // process the base 10..
        vector< int> result;
        if ( digits.size() <= 0) {
            return result;
        }
        int carry_bit = 0;
        // process the last bit
        int add_result = digits[ digits.size() - 1] + 1;
        if (add_result >= 10) {
            carry_bit = 1;
            digits[ digits.size() - 1] = add_result - 10;
        }
        else {
            digits[ digits.size() - 1] = add_result;
        }
        // process the other bits.
        for ( int i = digits.size() - 2; i >= 0; i--) {
            if (carry_bit == 0) {
                break;
            }
            int tmp = digits[i] + carry_bit;
            if (tmp >= 10) {
                carry_bit = 1;
                digits[i] = tmp - 10;
            }
            else {
                carry_bit = 0;
                digits[i] = tmp;
            }
        }
        // process the first bit
        if (carry_bit == 1) {
            result.push_back(1);
            result.insert(result.begin() + 1, digits.begin(), digits.end());
        }
        else {
            result.assign( digits.begin(), digits.end());
        }
        return result;
    }
};
int main() {
    Solution so;
    vector< int> vi;

}
