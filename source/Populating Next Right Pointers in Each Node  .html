Populating Next Right Pointers in Each Node

 
AC Rate: 1361/4053
My Submissions

Given a binary tree

    struct TreeLinkNode {
      TreeLinkNode *left;
      TreeLinkNode *right;
      TreeLinkNode *next;
    }

Populate each next pointer to point to its next right node. If there is no next right node, the next pointer should be set to NULL.

Initially, all next pointers are set to NULL.

Note:

  • You may only use constant extra space.
  • You may assume that it is a perfect binary tree (ie, all leaves are at the same level, and every parent has two children).

For example,
Given the following perfect binary tree,

         1
       /  \
      2    3
     / \  / \
    4  5  6  7

After calling your function, the tree should look like:

         1 -> NULL
       /  \
      2 -> 3 -> NULL
     / \  / \
    4->5->6->7 -> NULL

/**
 * Definition for binary tree with next pointer.
 * struct TreeLinkNode {
 *  int val;
 *  TreeLinkNode *left, *right, *next;
 *  TreeLinkNode(int x) : val(x), left(NULL), right(NULL), next(NULL) {}
 * };
 */
class Solution {
public:
    void connect(TreeLinkNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( root == NULL) {
            return;
        }
        TreeLinkNode *level = root;
        while (level) {
            TreeLinkNode *pre = NULL;
            TreeLinkNode *next = NULL;
            for (TreeLinkNode *p = level; p != NULL; p = p->next) {
                if (next == NULL) {
                    next = p->left ? p->left : p->right;
                }
                if (p->left) {
                    if (pre) {
                        pre->next = p->left;
                    }
                    pre = p->left;
                }
                if (p->right) {
                    if (pre) {
                        pre->next = p->right;
                    }
                    pre = p->right;
                }
            }
            level = next;
        }
       
    }
};
/**
 * Definition for binary tree with next pointer.
 * struct TreeLinkNode {
 *  int val;
 *  TreeLinkNode *left, *right, *next;
 *  TreeLinkNode(int x) : val(x), left(NULL), right(NULL), next(NULL) {}
 * };
 */
class Solution {
public :
    void connect(TreeLinkNode * root) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // level traversal, keep a global index, if reach 2^h - 1, add the NULL node.
        if ( root == NULL) {
            return;
        }
        // only one node, just return.
        if ( root->left == NULL && root->right == NULL) {
            return;
        }
        queue<TreeLinkNode*> level;
        AddToQueue(level, root);
        int index = 1;   //already has one
        int height = 2;  //start from the second level

        while (!level.empty()) {
            TreeLinkNode *pfirst = level.front();
            level.pop();
            index++;
            // add to queue
            AddToQueue(level, pfirst);

            while (index < (1 << height) - 1) {
                TreeLinkNode *pnext = level.front();
                level.pop();
                index++;
                // add to queue
                AddToQueue(level, pnext);

                pfirst->next = pnext; //update
                pfirst = pnext;
            }
            // process the node.
            if (index == (1 << height) - 1) {
                height++; //next level
            }
        }
           
    }
    void AddToQueue(queue<TreeLinkNode*> & q, TreeLinkNode * pnode) {
        if ( pnode->left != NULL) {
            q.push( pnode->left);
        }
        if ( pnode->right != NULL) {
            q.push( pnode->right);
        }
    }
};
