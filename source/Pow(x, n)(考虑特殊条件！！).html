Implement pow(x, n).


class Solution {
public:
    double pow(double x, int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 第一次出错因为没有考虑负数，第二次因为没有考虑负数最小值溢出 -2^31
        bool negative = false ;
        unsigned int un = n;
        if ( n < 0) {
            negative = true;
            un = ~( n - 1); //减一取反
        }
        double result = 1; //bug one negative
        while (un) {
            if (un & 1) {
                result = result * x;
            }
            x *= x;
            un >>= 1;
        }
        if (negative) {
            result = 1 / result;
        }
        return result;
    }
};
