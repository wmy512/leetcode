Implement regular expression matching with support for '.' and '*'.

'.' Matches any single character.
'*' Matches zero or more of the preceding element.

The matching should cover the entire input string (not partial).

The function prototype should be:
bool isMatch(const char *s, const char *p)

Some examples:
isMatch("aa","a") → false
isMatch("aa","aa") → true
isMatch("aaa","aa") → false
isMatch("aa", "a*") → true
isMatch("aa", ".*") → true
isMatch("ab", ".*") → true
isMatch("aab", "c*a*b") → true
#include <iostream>
using namespace std;
#include <vector>

class Solution {
public:
    bool isMatch(const char *s, const char *p) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.   
        if ( s == NULL || p == NULL) {
            return false;
        }
        if (!strcmp( p, ".*")) {
            return true;
        }
        return isMatchCore( s, p);
    }
    bool isMatchCore( const char * s, const char * p) {
        //一开始检测*(p+1)就好处理许多了
        if (* p == 0) {
            return * s == 0;
        }
        if (*( p + 1) != '*') {
            if ((* p == * s || * p == '.') && * s != 0) {
                return isMatchCore( s + 1, p + 1);
            }
            return false;
        }
        else { //*(p + 1) == '*'
            while ((* p == * s || * p == '.') && * s != 0) {
                if (isMatchCore( s, p + 2)) { //还是最小贪婪匹配
                    return true;
                }
                s++; //0 or more
            }
            return isMatchCore( s, p + 2);
        }
    }

};
int main() {
    Solution so;

    cout << so.isMatch( "aa", "aa" ) << endl;
    cout << so.isMatch( "aa", ".." ) << endl;
    cout << so.isMatch( "aa", ".*" ) << endl;
    cout << so.isMatch( "aaa", "aa" ) << endl;
    cout << so.isMatch( "aaa", "aa*" ) << endl;
    cout << so.isMatch( "a", "ab*" ) << endl;
    cout << so.isMatch( "aab", "c*a*b" ) << endl;
    cout << so.isMatch( "a", "ab*a" ) << endl;
    cout << so.isMatch( "aaa", "ab*a*c*a" ) << endl;

    return 0;
}
