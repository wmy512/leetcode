Follow up for "Remove Duplicates":
What if duplicates are allowed at most twice?

For example,
Given sorted array A = [1,1,1,2,2,3],

Your function should return length = 5, and A is now [1,1,2,2,3].

#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#include <string>

class Solution {
public :
    int removeDuplicates( int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( A == NULL || n <= 0) {
            return 0;
        }
        int first = 0;
        int cur = 0;
        int count = 1;
        while (++cur < n) {
            if ( A[cur] != A[first]) {
                A[++first] = A[cur];
                count = 1;  //重新计数
            }
            else {
                if (++count <= 2) {
                    A[++first] = A[cur];  ////更新下标 //bug [1, 1, 1, 1, 3, 3]
                }
            }
        }
        return first + 1;
    }
};
int main () {
    Solution so;
   
    int a[] = {1, 1, 1, 1, 3, 3};
    int n = so.removeDuplicates(a, 6);
    cout << n << endl;
    for ( int i = 0; i < 6; i++) {
        cout << a[i] << "  " ;
    }
    cout << endl;

   
    return 0;
}
