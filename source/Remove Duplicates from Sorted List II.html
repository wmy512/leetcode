Given a sorted linked list, delete all nodes that have duplicate numbers, leaving only distinct numbers from the original list.

For example,
Given 1->2->3->3->4->4->5, return 1->2->5.
Given 1->1->1->2->3, return 2->3.


#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>
#include <string>
#include <unordered_set>


 struct ListNode {
     int val;
     ListNode *next;
     ListNode(int x) : val(x), next(NULL) {}
 };
 
class Solution {
public:
    ListNode *deleteDuplicates( ListNode * head) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路，计数，数量大于1的删除,然后重建链表
        // 第一次出错，不做删除节点操作（可能不是堆上创建的）
        if ( head == NULL) {
            return 0;
        }
        ListNode *cur = head;
        ListNode *result_head = NULL;
        ListNode *result_tail = NULL;
        while (cur) {
            int count = 1;
            ListNode *next = cur->next;
            while (next && next->val == cur->val) {
                count++;
                next = next->next;
            }
            if (count == 1) {
                //build the linked list
                if (result_head == NULL) {
                    result_head = cur;
                }
                else {
                    result_tail->next = cur;
                }
                result_tail = cur;
            }
            else {
                //delete all the node //可能不是new 的，所以这里不做删除操作
                /*ListNode *ptmp = cur;
                while (cur && count-- >= 0) {
                    ptmp = cur;
                    delete ptmp;
                    cur = cur->next;
                }*/
            }
            cur = next;
        }
        if (result_tail) {
            result_tail->next = NULL;
        }
        return result_head;
    }
};
int main() {
    Solution so;
   
    ListNode h(1), hn(1), hnn(2);
    h.next = &hn;
    hn.next = &hnn;
    ListNode *result = so.deleteDuplicates(&h);
    while (result) {
        cout << result->val << " ";
        result = result->next;
    }
    return 0;
}
