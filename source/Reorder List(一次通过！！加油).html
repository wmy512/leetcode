Reorder List

Given a singly linked list L: L0→L1→…→Ln-1→Ln,
reorder it to: L0→Ln→L1→Ln-1→L2→Ln-2→…

You must do this in-place without altering the nodes' values.

For example,
Given {1,2,3,4}, reorder it to {1,4,2,3}.

Discuss


#include <iostream>

 //* Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

class Solution {
public:
    void reorderList( ListNode * head) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 先将链表拆分为两个，先获得长度，然后从一半处分开（偶数情况和奇数情况一致，比如 1 2 | 3 4， 1 2 3 | 4 5）(n + 1) / 2
        int length = 0;
        ListNode *cur = head;
        while (cur) {
            cur = cur->next;
            length++;
        }
        cur = head;
        int mid_length = (length + 1) / 2;
        int index = 0;
        ListNode *first_head = NULL, *first_tail = NULL;
        ListNode *second_head = NULL, *second_tail = NULL;
        while (cur) {
            ListNode *next = cur->next;
            cur->next = NULL;
            //first link list
            if (index++ < mid_length) {
                if (first_head == NULL) {
                    first_head = cur;
                }
                else {
                    first_tail->next = cur;
                }
                first_tail = cur;
            }
            else { //second link list
                if (second_tail == NULL) {
                    second_tail = cur;
                }
                else {
                    cur->next = second_head;
                }
                second_head = cur;
            }
            cur = next;
        }
        //组合链表
        ListNode *p1 = first_head;
        ListNode *p2 = second_head;
        while (p1) {
            ListNode *p1_next = p1->next;
            if (p2 == NULL) {
                break;
            }
            ListNode *p2_next = p2->next;
            p2->next = p1_next;
            p1->next = p2;
            p1 = p1_next;
            p2 = p2_next;
        }
    }
};
