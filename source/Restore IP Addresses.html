Given a string containing only digits, restore it by returning all possible valid IP address combinations.

For example:
Given "25525511135",

return ["255.255.11.135", "255.255.111.35"]. (Order does not matter)

Discuss


class Solution {
public:
    vector<string> restoreIpAddresses( string s) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 第一次出错因为010010 分解0出现问题
        vector< string> result;
        string one_s;
        dfs(result, s, one_s, 0, 0);
        return result;
    }
    void dfs(vector<string> & result, string s, string & one_s, int ip_index, int s_index) {
        if ( s_index >= s.size() && ip_index == 4) {
            result.push_back( one_s.substr(0, one_s.size() - 1)); //trim the last .
        }
        if ( ip_index < 4 && s_index < s.size()) {
            int num = 0;

            for ( int i = s_index; i < s.size() && i < s_index + 3; i++) {
               
                num = num * 10 + ( s[i] - '0');
                if (i > s_index && s[ s_index] == '0') { //对0进行过滤
                    continue;
                }
                //一个优化,如果剩余的位数过多，那么直接过滤
                if ( s.size() - i > (4 - ip_index) * 3) {
                    continue;
                }
               
                string origin_one = one_s;
                if (num <= 255) {
                    one_s += s.substr( s_index, i - s_index + 1) + ".";
                    //cout << one_s << endl;
                    ip_index++;
                    dfs( result, s, one_s, ip_index, i + 1);
                    ip_index--;
                    one_s = origin_one;
                }
            }
        }
    }
};
