Reverse digits of an integer.

Example1: x = 123, return 321
Example2: x = -123, return -321

Have you thought about this?

Here are some good questions to ask before coding. Bonus points for you if you have already thought through this!

If the integer's last digit is 0, what should the output be? ie, cases such as 10, 100.

Did you notice that the reversed integer might overflow? Assume the input is a 32-bit integer, then the reverse of 1000000003 overflows. How should you handle such cases?

Throw an exception? Good, but what if throwing an exception is not an option? You would then have to re-design the function (ie, add an extra parameter).
#include <iostream>
using namespace std;

class Solution {
public:
    int reverse(int x) {
        const int MAX_INT = ( int)(( unsigned)(~0) >> 1);
        const int MAX_DIV = MAX_INT / 10;
        const int MAX_REM = MAX_INT % 10;

        const int MIN_INT = ( int)((( unsigned)(~0) >> 1) + 1); //-2147483648
        const int MIN_DIV = ( unsigned)MIN_INT / 10; // 214748364
        const int MIN_REM = ( unsigned)MIN_INT % 10; // 8
        int result = 0;
        bool is_minus = x & 0x80000000;
       
        int unsigned_value = x;
        if (is_minus == true) {
            unsigned_value = ~( x - 1);
        }
        while (unsigned_value) {
            int rem = unsigned_value % 10;
            if (is_minus == true) { //is minus
                if (result > MIN_DIV || (result == MIN_DIV && rem > MIN_REM)) {
                    result = MIN_INT;
                    break;
                }
            }
            else {
                if (result > MAX_DIV || (result == MAX_DIV && rem > MAX_REM)) {
                    result = MAX_INT;
                    break;
                }
            }
            result = result * 10 + rem;
            unsigned_value /= 10;
        }
        if (is_minus == true) {
            return -result;
        }
        return result;

    }
};
int main() {
    Solution so;
    cout << so.reverse(10) << endl;
    cout << so.reverse(-100) << endl;
    cout << so.reverse(-123) << endl;
    cout << so.reverse(-2147483648) << endl; //overflow //0
    cout << so.reverse(2147483647) << endl;
    cout << so.reverse(1000000003) << endl;
    cout << so.reverse(-1000000003) << endl; //overflow //0
    return 0;
}
