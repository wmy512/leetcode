Reverse Nodes in k-Group

Given a linked list, reverse the nodes of a linked list k at a time and return its modified list.

If the number of nodes is not a multiple of k then left-out nodes in the end should remain as it is.

You may not alter the values in the nodes, only nodes itself may be changed.

Only constant memory is allowed.

For example,
Given this linked list: 1->2->3->4->5

For k = 2, you should return: 2->1->4->3->5

For k = 3, you should return: 3->2->1->4->5

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <string>



// Definition for singly-linked list.
 struct ListNode {
     int val;
     ListNode *next;
     ListNode(int x) : val(x), next(NULL) {}
 };

class Solution {
public:
    ListNode *reverseKGroup( ListNode * head, int k) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // first get the length of the list
        // 第一次出错，没有考虑length < k的情况
        if ( head == NULL || k <= 1) {
            return head;
        }
        int list_length = 0;
        ListNode *cur = head;
        while (cur) {
            cur = cur->next;
            list_length++;
        }
        cur = head;
        ListNode *result_head = NULL;
        ListNode *result_tail = NULL;
        ListNode *next = NULL;
        while (list_length >= k) {
           
            next = cur->next;
            int index = 0;
            //逆序插入k个元素，保留头节点和尾节点。//每组k个节点都逆序插入，最后连接
            ListNode *tmp_head = NULL;
            ListNode *tmp_tail = NULL;

            while (index < k) {
                next = cur->next;
                cur->next = NULL; //避免最后的不为空
                if (tmp_head == NULL) {
                    tmp_head = cur;
                    tmp_tail = cur;
                }
                else {
                    cur->next = tmp_head;
                    tmp_head = cur;
                }
                cur = next;
                index++;
            }

            //重建最后的链表 顺序插入
            if (result_head == NULL) {
                result_head = tmp_head;
                result_tail = tmp_tail;
            }
            else {
                result_tail->next = tmp_head;
                result_tail = tmp_tail;
            }
            list_length -= k;
        }
        if (list_length > 0) {
            //重建最后的链表 顺序插入
            if (result_head == NULL) {  //length < k
                result_head = head;
            }
            else {
                result_tail->next = next;
            }
        }
        return result_head;
    }
};
int main() {
    Solution so;

    ListNode h(1), h1(2), h2(3), h3(4), h4(5), h5(6);
    //h.next = &h1;
    h1.next = &h2;
    h2.next = &h3;
    h3.next = &h4;
    h4.next = &h5;

    ListNode *result = so.reverseKGroup(&h, 2);
    while (result) {
        cout << result->val << " ";
        result = result->next;
    }
    return 0;
}
