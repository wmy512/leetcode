Given a list, rotate the list to the right by k places, where k is non-negative.

For example:
Given 1->2->3->4->5->NULL and k = 2,
return 4->5->1->2->3->NULL.
class Solution {
public :
    ListNode *rotateRight(ListNode * head, int k) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 直接找到第k个点，然后移动下就可以了。
        if ( head == NULL) {
            return head;
        }
        int length = 0;
        ListNode *cur = head;
        while (cur) {
            length++;
            cur = cur->next;
        }
        k = k % length;
        if ( k == 0) {
            return head;
        }
        int first_length = length - k;
       
        ListNode *first_tail = head;
        while (first_length - 1 > 0) {
            first_tail = first_tail->next;
            first_length--;
        }
        ListNode *result_head = first_tail->next;
        first_tail->next = NULL;
        cur = result_head;
        while (cur->next) {
            cur = cur->next;
        }
        cur->next = head;
        return result_head;
       
    }
   
};

class Solution {
public :
    ListNode *rotateRight(ListNode * head, int k) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 经典字符串旋转算法，先各自逆序，然后整体逆序即可
        if ( head == NULL) {
            return head;
        }
        int length = 0;
        ListNode *cur = head;
        while (cur) {
            length++;
            cur = cur->next;
        }
        k = k % length;
        int first_length = length - k;
        ListNode *first_tail = head;
        while (first_tail && --first_length >= 0) {
            first_tail = first_tail->next;
        }
        ListNode *first_reverse_tail = head;
        ListNode *first_reverse_head = ReverseList( head, first_tail);
        ListNode *second_reverse_head = ReverseList(first_tail, NULL);
        first_reverse_tail->next = second_reverse_head;
        return ReverseList(first_reverse_head, NULL);
       
    }
    ListNode *ReverseList(ListNode * head, ListNode * tail) {
        //逆序链表并建立新链表
        ListNode *cur = head;
        ListNode *next = NULL;
        ListNode *reverse_head = NULL;
        while (cur != tail) {
            next = cur->next;
            cur->next = reverse_head;
            reverse_head = cur;
            cur = next;
        }
        return reverse_head;
    }
};
