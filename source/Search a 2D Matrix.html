Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties:

  • Integers in each row are sorted from left to right.
  • The first integer of each row is greater than the last integer of the previous row.

For example,

Consider the following matrix:

[
  [1,   3,  5,  7],
  [10, 11, 16, 20],
  [23, 30, 34, 50]
]

Given target = 3, return true.

//第一次给写成先按矩阵第一列二分搜索，导致[[1,3]], 3结果出错，改为最后一列搜索通过
#include <iostream>
using namespace std;
#include <vector>
#include <stack>
#include <queue>

class Solution {
public:
    bool searchMatrix( vector< vector< int> > & matrix, int target) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // two binary search
        if ( matrix.size() <= 0) {
            return false;
        }
        vector< int> search( matrix.size());
        for ( int i = 0; i < matrix.size(); i++) {
            search[i] = matrix[i][ matrix[i].size() - 1]; // fill the last element
        }
        int pos = LowerBound(search, target);
        if ( matrix[pos][0] == target) {
            return true;
        }
        //search target in the current row
        int pos_again = LowerBound( matrix[pos], target);
        return matrix[pos][pos_again] == target;

    }
    int LowerBound( vector< int> & vi, int target) {
        int low = 0;
        int high = vi.size() - 1;
        while (low < high) {
            int mid = ((high - low) >> 1) + low;
            if ( vi[mid] < target) {
                low = mid + 1;
            }
            else {
                high = mid;
            }
        }
        return low;
    }

};
