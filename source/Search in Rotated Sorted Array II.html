Write a function to determine if a given target is in the array.
#include <iostream>
#include <stack>
using namespace std;


class Solution {
public :
    bool search( int A[], int n, int target) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( A == NULL || n <= 0) {
            return -1;
        }
        int left = 0;
        int right = n - 1;
        while (left <= right) {   ////bug 1 left than or equal
            int mid = ((right - left) >> 1) + left;
            if ( A[mid] == target || A[left] == target || A[right] == target) {
                return true;
            }
            if ( A[mid] > A[left]) {        //确定左半部分
                if ( A[mid] < target || target < A[left]) {  //肯定在右半部分
                    left = mid + 1;
                }
                else {
                    right = mid - 1;
                }
            }
            else if ( A[mid] < A[left]) {  //确定右半部分
                if ( A[right] < target || target < A[mid]) { //肯定在左半部分
                    right = mid - 1;
                }
                else {
                    left = mid + 1;
                }
            }
            else {  //不确定，那么排除掉第一个元素
                left++;
            }
        }
        return false;
    }
};


int main() {
    Solution so;
   
    int a[] = {1, 3, 1, 1, 1};
    cout << so.search(a, 5, 3) << endl;
    return 0;
}
