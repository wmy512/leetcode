Given a m x n matrix, if an element is 0, set its entire row and column to 0. Do it in place.

Follow up:

Did you use extra space?
A straight forward solution using O(mn) space is probably a bad idea.
A simple improvement uses O(m + n) space, but still not the best solution.
Could you devise a constant space solution?


class Solution {
public:
    void setZeroes( vector< vector< int> > & matrix) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // use space (m + n)
       
        int m = matrix.size();
        int n = matrix[0].size();
        if (m <= 0 || n <= 0) {
            return;
        }
        vector< int> vn(n);
        vector< int> vm(m);
        for ( int i = 0; i < m; i++) {
            for ( int j = 0; j < n; j++) {
                if ( matrix[i][j] == 0) {
                    vm[i] = 1;
                    vn[j] = 1;
                }
            }
        }
        for ( int i = 0; i < m; i++) {
            for ( int j = 0; j < n; j++) {
                if (vm[i] || vn[j]) {
                    matrix[i][j] = 0;
                }
            }
        }
    }
};

class Solution {
public:
    void setZeroes( vector< vector< int> > & matrix) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // use space O(1)
        // 思路：先找到一个0元素，这样它的行列就可以当做临时的行和列了。
        int m = matrix.size();
        int n = matrix[0].size();
        if (m <= 0 || n <= 0) {
            return;
        }
        int pos_x = -1, pos_y = -1;
        for ( int i = 0; i < m; i++) {
            for ( int j = 0; j < n; j++) {
                if (pos_x != -1 && (i == pos_x || j == pos_y)) { //忽略掉那一行和列。。。
                    continue;
                }
                if ( matrix[i][j] == 0) {
                    if (pos_x == -1) {
                        pos_x = i;
                        pos_y = j;
                       
                        // 把其它不是0的值变为0，否则会和1冲突
                        for ( int k = 0; k < n; k++) {
                            matrix[pos_x][k] = matrix[pos_x][k] == 0 ? 1 : 0;
                        }
                        for ( int k = 0; k < m; k++) {
                            matrix[k][pos_y] = matrix[k][pos_y] == 0 ? 1 : 0;
                        }
                        matrix[pos_x][pos_y] = 1; //第j列为0,第i行为0

                    }
                    else {
                        matrix[pos_x][j] = 1; //第j列为0
                        matrix[i][pos_y] = 1; //第i行为0
                    }
                }
            }
        }
       
        if (pos_x == -1) { // bug one: no zero element
            return ;
        }
        for ( int i = 0; i < m; i++) {
            for ( int j = 0; j < n; j++) {
                if (i == pos_x || j == pos_y) {
                    continue;
                }
                if ( matrix[pos_x][j] || matrix[i][pos_y]) {
                    matrix[i][j] = 0;
                }
            }
        }
       
        for ( int i = 0; i < n; i++) {
            matrix[pos_x][i] = 0;
        }
        for ( int i = 0; i < m; i++) {
            matrix[i][pos_y] = 0;
        }
    }
};
