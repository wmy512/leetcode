Given an absolute path for a file (Unix-style), simplify it.

For example,
path = "/home/", => "/home"
path = "/a/./b/../../c/", => "/c"

click to show corner cases.

Corner Cases:

  • Did you consider the case where path = "/../"?
    In this case, you should return "/".
  • Another corner case is the path might contain multiple slashes '/' together, such as "/home//foo/".
    In this case, you should ignore redundant slashes and return "/home/foo".

                int tmp_i = i;
                while (tmp_i < path.size() && path[tmp_i] == '/') {
                    tmp_i++;
                }
                i += tmp_i - i - 1;
                continue;
            }
            filter += path[i];
            i++;
        }

        path = filter;
        string result;
        for ( int i = 0; i < path.size(); ) {
            if ( path.substr(i, 3) == "../") { //
                int first_slash = result.find_last_of( '/');
                if (first_slash != string::npos) {
                    int sec_slash = result.rfind( '/', first_slash - 1);
                    if (sec_slash != string::npos) {
                        result = result.substr(0, sec_slash + 1);
                    }
                }
                i += 3;
                continue;

            }
            if ( path.substr(i, 2) == "./") {
                i += 2;
                continue;
            }
           
           
           
            result += path[i];
            i++;

        }
        if (result == "/" || result == "/." || result == "/..") {
            return "/";
        }
       
        if (result.size() > 2 && result.substr(result.size() - 2, 2) == "..") {
            int first_slash = result.find_last_of( '/');
            if (first_slash != string::npos) {
                int sec_slash = result.rfind( '/', first_slash - 1);
                if (sec_slash != string::npos) {
                    result = result.substr(0, sec_slash + 1);
                }
            }
        }
        if (result.size() > 1 && result[result.size() - 1] == '.') {
            result = result.substr(0, result.size() - 2);
        }
        if (result.size() > 1 && result[result.size() - 1] == '/') {
            result = result.substr(0, result.size() - 1);
        }
        return result;

    }

class Solution {
public:
    string simplifyPath( string path) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        //用栈去储存每对"//"之间的东西，遇到"/"可以直接忽略，其它的如果是".."那么弹出一个，如果不是"."那么可以直接插入
        vector< string> pure_path;
        int i = 0;
        while (i < path.size()) {
            //skip the '////'
            while ( path[i] == '/' && i < path.size()) {
                i++;
            }
            if (i == path.size()) {
                break;
            }
            int left = i;
            int right = i;
            while (right < path.size() && path[right] != '/') {
                right++;
            }
            string part = path.substr(i, right - i);
            if (part == "..") {
                if (pure_path.size() > 0) {
                    pure_path.pop_back();
                }
            }
            else if (part != ".") {  //'.'直接忽略
                pure_path.push_back(part);
            }
            i = right;
        }
        string result;
        if (pure_path.size() == 0) {
            result = "/";
        }
        else {
            for ( int i = 0; i < pure_path.size(); i++) {
                result = result + "/" + pure_path[i];
            }
        }
        return result;
    }
};
