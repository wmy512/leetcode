Given an array of integers, every element appears three times except for one. Find that single one.

Note:
Your algorithm should have a linear runtime complexity. Could you implement it without using extra memory?

#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#include <string>

class Solution {
public:
   int singleNumber( int A [], int n ) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 变成三进制存储，依次累加，溢出不进位，最后的值再转化为数值即可（可找到只出现一次）
        // 变成三进制存储，依次乘以2累加，溢出不进位，最后的值再转化为数值即可（可找到只出现两次的那个数）
        // 第一次出了bug，没有处理负数
        // 第二次bug，负数 -2^31 变成正数溢出，换成unsigned int
        if ( A == NULL || n <= 0) {
            return 0;  //error code, not known
        }
        int three_bit[30] = {0};
        int negative_num = 0;
        for ( int i = 0; i < n; i++) {
            unsigned int value = A[i];
          
            int j = 0;
            if ( A[i] <= 0) {
                negative_num++;
                value = ~( A[i] - 1);
            }
            while (value) {
                three_bit[j++] += value % 3;
                value /= 3;
            }
        }
        int max_bit = 29;
      
        for ( int i = 0; i < max_bit; i++) {
            three_bit[i] %= 3;
        }
        while (max_bit >= 0 && three_bit[max_bit] == 0) {
            max_bit--;
        }
        int result = 0;
        while (max_bit >= 0) {
            result = result * 3 + three_bit[max_bit];
            max_bit--;
        }
        if (negative_num % 3 != 0) { //process negative
            result = -result;
        }
        return result;
    }
};
