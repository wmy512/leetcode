Given an array with n objects colored red, white or blue, sort them so that objects of the same color are adjacent, with the colors in the order red, white and blue.

Here, we will use the integers 0, 1, and 2 to represent the color red, white, and blue respectively.

Note:
You are not suppose to use the library's sort function for this problem.

click to show follow up.

Follow up:
A rather straight forward solution is a two-pass algorithm using counting sort.
First, iterate the array counting number of 0's, 1's, and 2's, then overwrite array with total number of 0's, then 1's and followed by 2's.

Could you come up with an one-pass algorithm using only constant space?

#include <iostream>
using namespace std;
#include <vector>
#include <algorithm>
#include <string>

class Solution {
public :
    void sortColors( int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // two-pass algorithm
        if ( A == NULL || n <= 0) {
            return;
        }
        int count_array[3] = {0};
        for ( int i = 0; i < n; i++) {
            count_array[ A[i]]++;
        }
        count_array[1] += count_array[0];
        count_array[2] += count_array[1];
        int i = 0, k = 0;
        while (i < n) {
            while (i < count_array[k]) {
                A[i++] = k;
            }
            k++;
        }
    }
    // one-pass algorithm
    void sortColors2( int A[], int n) {
        if ( A == NULL || n <= 0) {
            return;
        }
        // 用r 表示0，g表示，cur是现在元素，如果是红色，放到r，r后移，如果是绿色，放到g，g前移，如果是白色，就是那个位置
        int r, b, cur;
        for (r = 0, b = n - 1, cur = 0; cur <= b;) {
            if ( A[cur] == 0) {
                swap( A[cur++], A[r++]); //类似快排，这部分可以一直向前走
            }
            else if ( A[cur] == 2) {
                swap( A[cur], A[b--]); //可能交换出红色或者绿色，需要再次判断
            }
            else {
                cur++; //白色也可以直接走
            }
        }
    }
};
int main () {
    Solution so;
   
    int a[] = {0, 1, 2, 2, 1, 0, 1, 1};
    so.sortColors2(a, 8);
    for ( int i = 0; i < 8; i++) {
        cout << a[i] << "  " ;
    }
    cout << endl;

   
    return 0;
}
