Implement int sqrt(int x).

Compute and return the square root of x.

class Solution {
public:
    int sqrt(int x) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 二分法查找这个值....
        // 第一次出错，没有考虑根是小数的，如2,而且要是用fabs！！
        // 第二次出错，比如3，开根号是1.732，但是一旦舍入之后就是2，这个结果不对，需要后处理
        // 第三次出错， 2147483647， 导致int平方后溢出....
        if ( x <= 0) { //不考虑负数
            return 0;
        }
        double left = 1, right = x;
        while (fabs(right - left) > 0.01) {
            double mid = left + (right - left) / 2;
            double mid_square = mid * mid;
           
            if (mid_square < x) {
                left = mid;
            }
            else if (mid_square > x) {
                right = mid;
            }
            else {
                return mid;
            }
            //cout << mid << " " << left << endl;
        }
        if (( double)( int)(left + 0.5) * ( double)( int)(left + 0.5) > x) { //避免溢出，再次强转为double
            return left;
        }
        return (left + 0.5); //四舍五入
    }
};
