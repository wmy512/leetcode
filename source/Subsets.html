Given a set of distinct integers, S, return all possible subsets.

Note:

  • Elements in a subset must be in non-descending order.
  • The solution set must not contain duplicate subsets.

For example,
If S = [1,2,3], a solution is:

[
  [3],
  [1],
  [2],
  [1,2,3],
  [1,3],
  [2,3],
  [1,2],
  []
]

#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>

struct TreeNode {
     int val;
     TreeNode *left;
     TreeNode *right;
     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 };

class Solution {
public:
    vector<vector<int> > subsets( vector< int> & S) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        vector< int> vi;
        vector< vector< int> > result;
        result.push_back(vi);
        int length = S.size();
        if (length <= 0) {
            return result;
        }
        sort( S.begin(), S.end()); //bug one..non-descending
        //用 二进制位组合可以得出所有的数据，也可以用dfs搜索
        int max_element = 1 << length;
        for ( int i = 1; i < max_element; i++) {
            vector< int> vtmp;
            int bit = 0;
            int itmp = i;
            while (itmp) {
                int num = itmp & 1;
                if (num == 1) {
                    vtmp.push_back( S[bit]);
                   
                }
                bit++;
                itmp >>= 1;
            }
            result.push_back(vtmp);
        }
        return result;

    }
};
int main() {
    Solution so;
   
    vector<int> s;
    for (int i = 1; i <= 3; i++) {
        s.push_back(i);
    }
    vector<vector<int> > result = so.subsets(s);
    for (int i = 0; i < result.size(); i++) {
        for ( int j = 0; j < result[i].size(); j++) {
            cout << result[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
