Substring with Concatenation of All Words

 Total Accepted: 1076 Total Submissions: 6977My Submissions

You are given a string, S, and a list of words, L, that are all of the same length. Find all starting indices of substring(s) in S that is a concatenation of each word in L exactly once and without any intervening characters.

For example, given:
S: "barfoothefoobarman"
L: ["foo", "bar"]

You should return the indices: [0,9].
(order does not matter).

class Solution {
public :
    vector< int> findSubstring( string S, vector< string> & L) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 直接遍历S，对每个字符开始的字符串就行切分，一旦直到所有单词全部匹配，
        // 1680ms
        vector< int> result;
        if ( L.size() == 0) {
            return result;
        }
        map< string, int> dict;
        map< string, int> cur;
        for ( int i = 0; i < L.size(); i++) {
            dict[ L[i]]++;
        }
        int m = L[0].size(), n = L.size();
        for ( int i = 0; i < S.size(); i++) {
            if (i + m * n > S.size()) { //剩余字符太少了
                break;
            }
            cur.clear();
            string part = S.substr(i, m * n);
            int j = 0;
            for (j = 0; j < n; j++) {
                string word = part.substr(j * m, m);
                if (dict.find(word) == dict.end()) {
                    break;
                }
                cur[word]++;
                if (cur[word] > dict[word]) {
                    break;
                }
            }
            if (j == n) {
                result.push_back(i);
            }
        }
        return result;
    }
};



class Solution {
public:
    vector<int> findSubstring( string S, vector< string> & L) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 利用min window思路，O(S.size() * L[0].size())复杂度，找到min_len = m * n的将位置保存就好了
        // 第一次把need内的元素搞错了
        // 492ms
        vector< int> result;
        if ( L.size() == 0) {
            return result;
        }
        map< string, int> dict;
        map< string, int> need;
        for ( int i = 0; i < L.size(); i++) {
            dict[ L[i]]++;
        }
        int min_len = INT_MAX, left = 0, right = 0;
       
        int m = L[0].size(), n = L.size();
        for ( int offset = 0; offset < m; offset++) {
        //int offset = 0;
            need.clear();
            int count = 0;
            for ( int i = offset, j = offset; i < S.size(); i += m) {  //完整的min_windows算法
                string word = S.substr(i, m);
                if (need[word] < dict[word]) {
                    count++;
                }
                need[word]++;
                if (count == n) {
                    while (j <= i && need[ S.substr(j, m)] > dict[ S.substr(j, m)]) {
                        need[ S.substr(j, m)]--;
                        j += m;
                    }
                    int tmp_len = i - j + m;
                    if (tmp_len == m * n && tmp_len <= min_len) {
                        left = j;
                        right = i;
                        min_len = tmp_len;
                        result.push_back(left);
                    }
                    if (j <= i) {
                        need[ S.substr(j, m)]--;
                        count--;
                        j += m;
                    }
                }
            }
        }

        return result;
    }
};
