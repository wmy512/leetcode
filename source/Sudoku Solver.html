Write a program to solve a Sudoku puzzle by filling the empty cells.

Empty cells are indicated by the character '.'.

You may assume that there will be only one unique solution.


A sudoku puzzle...


...and its solution numbers marked in red.

Discuss

#include <iostream>
using namespace std;
#include <vector>


class Solution {
private:
    static const int N = 9;
public:
    void solveSudoku( vector< vector< char> > & board) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
       
        int rows[N][N + 1] = {0};
        int cols[N][N + 1] = {0};
        int blocks[N][N + 1] = {0};

        //init the value
        for ( int i = 0; i < board.size(); i++) {
            for ( int j = 0; j < board[i].size(); j++) {
                if ( board[i][j] != '.') {
                    int num = board[i][j] - '0';
                    rows[i][num] = 1;
                    cols[j][num] = 1;
                    blocks[i / 3 * 3 + j / 3][num] = 1;
                }
            }
        }
        //answer the value
        dfs( board, rows, cols, blocks);
        PrintVector( board);
    }
    bool dfs(vector<vector<char> > & board, int rows[][N + 1], int cols[][N + 1], int blocks [][N + 1]) {
        for ( int i = 0; i < board.size(); i++) {
            for ( int j = 0; j < board[i].size(); j++) {
                if ( board[i][j] == '.') {
                    for ( int num = 1; num <= 9; num++) {
                        if ( rows[i][num] || cols[j][num] || blocks[i / 3 * 3 + j / 3][num]) { //already used
                            continue;
                        }
                        board[i][j] = num + '0';
                        rows[i][num] = 1;
                        cols[j][num] = 1;
                        blocks[i / 3 * 3 + j / 3][num] = 1;
                        if (dfs( board, rows, cols, blocks)) {
                            return true;
                        }
                        board[i][j] = '.';
                        rows[i][num] = 0;
                        cols[j][num] = 0;
                        blocks[i / 3 * 3 + j / 3][num] = 0;
                    }
                    return false; // 进入这里，说明后面的结果都是错的
                }
            }
        }
        return true; //没有空余了
       
    }
    void PrintVector( vector< vector< char> > & board) {
        for ( int i = 0; i < board.size(); i++) {
            for ( int j = 0; j < board[i].size(); j++) {
                cout << board[i][j] << " ";
            }
            cout << endl;
        }
    }
};
int main() {
    Solution so;
    char *sudo[] = { "53..7....",
                    "6..195...",
                    ".98....6.",
                    "8...6...3",
                    "4..8.3..1",
                    "7...2...6",
                    ".6....28.",
                    "...419..5",
                    "....8..79",
    };
    vector<vector<char> > sudoku;
    for (int i = 0; i < 9; i++) {
        vector< char> tmp(sudo[i], sudo[i] + 9);
        sudoku.push_back(tmp);
    }
    so.PrintVector(sudoku);
    so.solveSudoku(sudoku);
    return 0;
}
