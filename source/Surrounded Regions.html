Surrounded Regions

Given a 2D board containing 'X' and 'O', capture all regions surrounded by 'X'.

A region is captured by flipping all 'O's into 'X's in that surrounded region .

For example,

X X X X
X O O X
X X O X
X O X X

After running your function, the board should be:

X X X X
X X X X
X X X X
X O X X

Discuss

#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <string>
#include <queue>
#include <cctype>


class Solution {
public:
    void solve(vector<vector<char>> & board) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：从四周出发，遇到'O'去遍历四个方向，遇到'O'就转变为'#'，然后知道无路可走。。这样所有的'#'最终都是能走到边界的
        // 而留下的'O'都不能走到边界
        if ( board.size() == 0) { //第一次出错没有处理边界
            return ;
        }
        for ( int i = 0; i < board.size(); i++) {
            dfs( board, i, 0);                      //left
            dfs( board, i, board[0].size() - 1);    //right
        }
        for ( int j = 1; j < board[0].size() - 1; j++) {
            dfs( board, 0, j);                      //up
            dfs( board, board.size() - 1, j);       //down
        }
        for ( int i = 0; i < board.size(); i++) {
            for ( int j = 0; j < board[0].size(); j++) {
                if ( board[i][j] == 'O') {
                    board[i][j] = 'X';
                }
                if ( board[i][j] == '#') {
                    board[i][j] = 'O';
                }
            }
        }
    }

    void dfs(vector<vector<char> > & board, int i, int j) {
        if ( i < 0 || i >= board.size() || j < 0 || j >= board[0].size()) {
            return;
        }
        if ( board[ i][ j] == 'O') {
            board[ i][ j] = '#';
            dfs( board, i - 1, j); //up
            dfs( board, i, j + 1); //right
            dfs( board, i + 1, j); //down
            dfs( board, i, j - 1); //left;
        }
    }
};
int main() {
    Solution so;
   
    vector<vector<char>> board;
    char *board_array[] = { "XXXX", "XOOX" , "XXOX" , "XOXX" };
    for (int i = 0; i < 4; i++) {
        vector< char> tmp(board_array[i], board_array[i] + 4);
        board.push_back(tmp);
    }
    so.solve(board);
    for (int i = 0; i < board.size(); i++) {
        for ( int j = 0; j < board[0].size(); j++) {
            cout << board[i][j] << " ";
        }
        cout << endl;
    }
    return 0;
}
