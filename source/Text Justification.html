Text Justification

Given an array of words and a length L, format the text such that each line has exactly L characters and is fully (left and right) justified.

You should pack your words in a greedy approach; that is, pack as many words as you can in each line. Pad extra spaces ' ' when necessary so that each line has exactly Lcharacters.

Extra spaces between words should be distributed as evenly as possible. If the number of spaces on a line do not divide evenly between words, the empty slots on the left will be assigned more spaces than the slots on the right.

For the last line of text, it should be left justified and no extra space is inserted between words.

For example,
words: ["This", "is", "an", "example", "of", "text", "justification."]
L: 16.

Return the formatted lines as:

[
   "This    is    an",
   "example  of text",
   "justification.  "
]

Note: Each word is guaranteed not to exceed L in length.

click to show corner cases.

Corner Cases:

  • A line other than the last line might contain only one word. What should you do in this case?
    In this case, that line should be left-justified.

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <string>

class Solution {
public:
    vector<string> fullJustify( vector< string> & words, int L) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 第一次出错，最后一行没有单独处理，单词中间只有一个空格，但是最后要填充足够的空格
        // 没有处理空单词
        // 第三次出错，中间的空格是按一半分割，不能全都给最左边
        vector< string> result;
        if ( words.size() == 0) {
            result.push_back(GetBlankString( L));
        }
        for ( int i = 0; i < words.size(); ) {
            string word = words[i];
            int len = word.size();
            int j = i + 1;
       
            while (j < words.size() && len + words[j].size() + (j - i) <= L) { //加上必要的空格数
                len += words[j].size();
                j++;
            }
            //结尾特殊处理
            if (j == words.size()) {
                string one = "";
                int tmp_len = 0;
                for ( int k = i; k < j - 1; k++) {
                    one += words[k] + " ";
                    tmp_len += words[k].size() + 1;
                }
                one += words[j - 1];
                tmp_len += words[j - 1].size();
                one += GetBlankString( L - tmp_len);
                result.push_back(one);
                break;
            }
            ////单词数为一和2时时单独处理
            if (j == i + 1) {
                int stand_num = L - len;
                string one = words[i] + GetBlankString(stand_num);
                result.push_back(one);
            }
            else if (j == i + 2) {
                int stand_num = L - len;
                string one = words[i] + GetBlankString(stand_num) + words[i + 1];
                result.push_back(one);
            }
            else {
                int stand_num = ( L - len) / (j - i - 1); //标准的空格数
                int left = stand_num + ( L - len - stand_num * (j - i - 1)); //不能整除，多余的给左边
                //转变为把一个数最均匀分给n个槽的问题
                int slot = j - i - 1;
                int total = L - len;
                vector< int> blank_num;
                GetSplit(blank_num, slot, total);
                string one = "";
                for ( int k = i; k < j - 1; k++) {
                    one += words[k] + GetBlankString(blank_num[k - i]);
                }
                one += words[j - 1];
                result.push_back(one);
            }
            i = j;
        }
        return result;
    }
    string GetBlankString( int num) {
        string result = "";
        for ( int i = 0; i < num; i++) {
            result += " ";
        }
        return result;
    }
    void GetSplit( vector< int> & result, int slot, int total) {
        //从后往前填充
        result.assign( slot, 0);
        int index = slot - 1;
        int num = total / slot;
        int residue = total;
        while (num * slot != residue && slot > 1) {
            result[index--] = num;
            residue = residue - num;
            num = (residue) / ( slot - 1);
            slot--;
        }
        if (num * slot == residue) {
            while (index >= 0) {
                result[index--] = num;
            }
        }
        else if ( slot == 1) {
            result[index--] = residue;
        }

    }

};
int main() {
    Solution so;
    char *word_arr[] = {"Don't","go" ,"around" ,"saying" ,"the" ,"world" ,"owes" ,"you" ,"a" ,"living;" ,"the" ,"world" ,"owes" ,"you" ,"nothing;" ,"it" ,"was" ,"here" ,"first." };
    vector<string> words;
    for (int i = 0; i < 19; i++) {
        words.push_back(word_arr[i]);
    }
    vector<string> result = so.fullJustify(words, 30);
    for (int i = 0; i < result.size(); i++) {
        cout << result[i] << endl;
    }
    vector<int> blank_num;
    so.GetSplit(blank_num, 4, 6);
    for (int i = 0; i < blank_num.size(); i++) {
        cout << blank_num[i] << " ";
    }
    cout << endl;
   
    return 0;
}
