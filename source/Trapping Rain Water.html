Given n non-negative integers representing an elevation map where the width of each bar is 1, compute how much water it is able to trap after raining.

For example, 
Given [0,1,0,2,1,0,1,3,2,1,2,1], return 6.


The above elevation map is represented by array [0,1,0,2,1,0,1,3,2,1,2,1]. In this case, 6 units of rain water (blue section) are being trapped. Thanks Marcos for contributing this image!

#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>

class Solution {
public:
    int trap(int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：遍历O(n*n)，找到每个点，左边比它高的，然后找到右边比它高的，min(max_right, max_left) - height 就是能储水容量
        // 耗时636ms
        if ( A == NULL || n <= 1) {  //第0个点怎么也不可能存水
            return 0;
        }
        int result = 0;
        for ( int i = 1; i < n - 1; i++) {
            int height = A[i];
            int li = i, ri = i;
            int max_left = height, max_right = height;
            //find greater than or equal left
            while (li >= 0) {
                if (max_left <= A[li]) {
                    max_left = A[li];
                    //break;
                }
                li--;
            }
            //find greater than or equal right
            while (ri < n) {
                if (max_right <= A[ri]) {
                    max_right = A[ri];
                    //break;
                }
                ri++;
            }
            result += min(max_left, max_right) - height;
        }
        return result;
    }
    int trap2(int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：遍历O(n*n)，找到每个点，左边比它高的，然后找到右边比它高的，min(max_right, max_left) - height 就是能储水容量
        // 优化：保存最大值和最小值，及其对应索引，如果下次的索引在其和最大小值之间，那么可以直接使用,降为48ms
        if ( A == NULL || n <= 1) {  //第0个点怎么也不可能存水
            return 0;
        }
        int result = 0;
        int max_left = A[1], max_right = A[1];
        int max_li = 1, max_ri = 1;
        for ( int i = 1; i < n - 1; i++) {
            int height = A[i];

            if (max_li < i && i < max_ri && height <= max_left && height <= max_right) {
                result += min(max_left, max_right) - height;
                continue;
            }
            int li = i, ri = i;
            int left = height, right = height;
            //find greater than or equal left
            while (li >= 0) {
                if (left <= A[li]) {
                    left = A[li];
                    max_left = left;
                    max_li = li;
                    //break;
                }
                li--;
            }
            //find greater than or equal right
            while (ri < n) {
                if (right <= A[ri]) {
                    right = A[ri];
                    max_right = right;
                    max_ri = ri;
                    //break;
                }
                ri++;
            }
            result += min(max_left, max_right) - height;
        }
        return result;
    }
    int trap3(int A[], int n) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 思路：申请O(N)的空间保存左边最大的高度，从左至右扫面填满。然后从右向左扫描，从而min(left, right)-height
        // 48ms （注意中间要保证min_bound大于0再进行存水）
        if ( A == NULL || n <= 2) {
            return 0;
        }
        vector< int> left_height( n);

        int trap_water = 0;
        int left_max = A[0];
        left_height[0] = left_max;
        for ( int i = 1; i < n; i++) {
            left_height[i] = left_max;
            left_max = left_max < A[i] ? A[i] : left_max;
        }
        int right_max = A[ n - 1];
        for ( int i = n - 2; i > 0; i--) {
            int min_bound = min(left_height[i], right_max);
            if (min_bound > A[i]) { //必须保证大于0
                trap_water += min_bound - A[i];
            }
            right_max = right_max < A[i] ? A[i] : right_max;
        }
        return trap_water;
    }
};
int main() {
    Solution so;
    int a[] = {0,1,0,2,1,0,1,3,2,1,2,1};
    cout << so.trap3(a, 12) << endl; //6
    return 0;
}
