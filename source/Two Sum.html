Given an array of integers, find two numbers such that they add up to a specific target number.

The function twoSum should return indices of the two numbers such that they add up to the target, where index1 must be less than index2. Please note that your returned answers (both index1 and index2) are not zero-based.

You may assume that each input would have exactly one solution.

Input: numbers={2, 7, 11, 15}, target=9
Output: index1=1, index2=2

Discuss


class Solution {
private:
    struct Node {
        int value;
        int index;
    };
public:
    vector<int> twoSum( vector< int> & numbers, int target) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 第一次出错，索引出错，建立Node改变
        // 第二次出错，没有按大小值排序结果
        vector< Node> node( numbers.size());
        for ( int i = 0; i < numbers.size(); i++) {
            node[i].value = numbers[i];
            node[i].index = i;
        }
        sort(node.begin(), node.end(), Compare);
        vector< int> result;
        for ( int low = 0, high = node.size() - 1; low < high; ) {
            int sum = node[low].value + node[high].value;
            if ( target < sum) {
                high--;
            }
            else if (sum < target) {
                low++;
            }
            else {
               
                result.push_back(min(node[low].index + 1, node[high].index + 1));
                result.push_back(max(node[low].index + 1, node[high].index + 1));
                return result;
            }
        }
    }
    static bool Compare(const Node &lhs, const Node &rhs) {
        return lhs.value < rhs.value;
    }

};
