Follow up for "Unique Paths":

Now consider if some obstacles are added to the grids. How many unique paths would there be?

An obstacle and empty space is marked as 1 and 0 respectively in the grid.

For example,

There is one obstacle in the middle of a 3x3 grid as illustrated below.

[
  [0,0,0],
  [0,1,0],
  [0,0,0]
]

The total number of unique paths is 2.

Note: m and n will be at most 100.


#include <iostream>
#include <stack>
using namespace std;
#include <algorithm>
#include <vector>
#include <cmath>

class Solution {
public:
    int uniquePathsWithObstacles( vector< vector< int> > & obstacleGrid) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // typical dynamic process vi[1][1] = 1; vvi[i][j] = vvi[i - 1][j] + vvi[i][j - 1];
        int m = obstacleGrid.size();
        if (m <= 0) {
            return 0;
        }
        int n = obstacleGrid[0].size();
        if (n <= 0) {
            return 0;
        }
        vector< vector< int> > vvi(m + 1);
        // init (m+1) * (n+1) matrix
        for ( int i = 0; i <= m; i++) {
            vector< int> vi(n + 1, 0);
            vvi[i] = vi;
        }
        vvi[1][1] = 1;
        for ( int i = 1; i <= m; i++) {
            for ( int j = 1; j <= n; j++) {
                if (i == 1 && j == 1) {
                    vvi[i][j] = 1;
                }
                else {
                    vvi[i][j] = vvi[i - 1][j] + vvi[i][j - 1];
                }
                if ( obstacleGrid[i - 1][j - 1] == 1) {   // processing Obstacles
                    vvi[i][j] = 0;
                }
            }
        }
        return vvi[m][n];   
    }
};
int main() {
    Solution so;

     // bug one [[1]] => 0
    int a[][3] = {0, 0, 0, 0, 1, 0, 0, 0, 0};
    vector<vector<int> > vvi(3);
    for (int i = 0; i < 3; i++) {
        vector< int> vi(a[i], a[i] + 3);
        vvi[i] = vi;
    }
    cout << so.uniquePathsWithObstacles(vvi) << endl; //1 bug one
    return 0;
}
