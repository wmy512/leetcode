Validate if a given string is numeric.

Some examples:
"0" => true
" 0.1 " => true
"abc" => false
"1 a" => false
"2e10" => true

Note: It is intended for the problem statement to be ambiguous. You should gather all requirements up front before implementing one.

Discuss



#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <string>
#include <queue>
#include <cctype>


class Solution {
public:
    bool isblank(int ch) {
        return ( ch == ' ' || ch == '\t' || ch == '\n');
    }
    bool isNumber( const char * s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 1. "0" => true
        // 2. " 0.1 " => true
        // 3. "abc" => false, NULL => false, "" => false
        // 4. "1 a" => false "1 ." => false
        // 5. "2e10" => true, "2e-10" => true, "2e+10" => true, "20.10e5" => true
        if ( s == NULL) {
            return false;
        }
        //trim
        string trim_str;
        const char *begin = s;
        const char *end = s + strlen( s);
        while (*begin) {
            if (!isblank(*begin)) {
                break;
            }
            begin++;
        }
        while (end > s) {
            --end;
            if (!isblank(*end)) {
                break;
            }
        }
        while (begin <= end) {
            trim_str += *begin;
            begin++;
        }
        //cout << trim_str << endl;
        //检查特殊字符的出现,顺便统计特殊字符的个数
        int char_map[256] = {0};
        for ( int i = 0; i < trim_str.size(); i++) {
            if (trim_str[i] != '.' && trim_str[i] != 'e' && trim_str[i] != '+' && trim_str[i] != '-') {
                if (trim_str[i] < '0' || trim_str[i] > '9') {
                    return false;
                }
            }
            char_map[trim_str[i]]++;
        }
        if (char_map[ '.'] > 1 || char_map[ 'e'] > 1) {
            return false;
        }
        if (trim_str.find( ".e") == 0 || trim_str.find("e." ) != -1) { //"1.e"是对的 但是".e"是错的
            return false;
        }
        // 去掉符号位
        if (trim_str == "") {
            return false;
        }
        if (trim_str[0] == '+' || trim_str[0] == '-') {
            trim_str = trim_str.substr(1); //rest
        }
        if (trim_str == "") {
            return false;
        }

        // 删除e+, e-, e
        int index0 = trim_str.find( '.');
        int index1 = trim_str.find( "e+");
        if (index1 != -1) {
            if (index1 == 0 || index1 + 2 == trim_str.size() || index0 > index1) {
                return false;
            }
            trim_str = trim_str.substr(0, index1) + trim_str.substr(index1 + 2, trim_str.size() - 2);
        }
        int index2 = trim_str.find( "e-");
        index0 = trim_str.find( '.');  //每次都要更新
        if (index2 != -1) {
            if (index2 == 0 || index2 + 2 == trim_str.size() || index0 > index2) {
                return false;
            }
            trim_str = trim_str.substr(0, index2) + trim_str.substr(index2 + 2, trim_str.size() - 2);
        }

        int index3 = trim_str.find( "e");
        index0 = trim_str.find( '.');
        if (index3 != -1) {
            if (index3 == 0 || index3 + 1 == trim_str.size() || index0 > index3) {
                return false;
            }
            trim_str = trim_str.substr(0, index3) + trim_str.substr(index3 + 1, trim_str.size() - 1);
        }
       
        index0 = trim_str.find( '.');
        if (index0 != -1) {
            // ".1" 和 "1."=> 被认为对的"
            trim_str = trim_str.substr(0, index0) + trim_str.substr(index0 + 1, trim_str.size() - 1);
        }

        if (trim_str == "") {
            return false;
        }
        //再检查一遍
        //应该全部是数字了
        for ( int i = 0; i < trim_str.size(); i++) {
            if (trim_str[i] < '0' || trim_str[i] > '9') {
                return false;
            }
        }
       

        return true;
    }
};
int main() {
    Solution so;
    cout << 2e0 << endl;
    cout << 1e1 << endl;
    cout << 4.e3 << endl;
    cout << so.isNumber( "           20.e59           " ) << endl; //0
   
    cout << so.isNumber( " 0 a") << endl; //0
    cout << so.isNumber( NULL) << endl; // 0
    cout << so.isNumber( "abc") << endl; // 0
    cout << so.isNumber( "e") << endl; // 0
    cout << so.isNumber( "") << endl; // 0
    cout << so.isNumber( "1e") << endl; // 0
    cout << so.isNumber( "2e0.6") << endl; // 0
    cout << so.isNumber( "2e0.6") << endl; // 0
    cout << so.isNumber( "2e-10-") << endl; //0
    cout << so.isNumber( ".") << endl; //0
    cout << so.isNumber( ".e3") << endl; //0

    cout << so.isNumber( "4.e3") << endl; //1
    cout << so.isNumber( "2.4e06") << endl; // 1
    cout << so.isNumber( "2e-10") << endl; //1
    cout << so.isNumber( " 0") << endl; //1
    cout << so.isNumber( "-2e-10") << endl; //1
    cout << so.isNumber( ".5") << endl; //1
    cout << so.isNumber( "0.") << endl; //1
   
    return 0;
}
