Given a string containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid.

The brackets must close in the correct order, "()" and "()[]{}" are all valid but "(]" and "([)]" are not.

#include <iostream>
#include <stack>

using namespace std;


class Solution {
public:
    bool isValid(string s) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        stack< int> si;
        if ( s.size() <= 0) {
            return false;
        }
        si.push( s[0]);
        bool result = true;
        for ( int i = 1; i < s.size(); i++) {
            if ( s[i] == ')') {
                result = ValidatePair(si, '(', ')');
            }
            else if ( s[i] == ']') {
                result = ValidatePair(si, '[', ']');
            }
            else if ( s[i] == '}') {
                result = ValidatePair(si, '{', '}');
            }
            else {
                si.push( s[i]);
            }
            if (result == false) {
                break;
            }
        }
           
        return result && si.empty();
    }
    bool ValidatePair( stack< int> & si, int prev, int now) {
        if (! si.empty()) {
            int top = si.top();
            si.pop();
            if (top != prev) {
                return false;
            }
            return true;
        }
        return false;
    }
};

int main() {
    Solution so;
    cout << so.isValid( "(([][()))") << endl;
