Implement wildcard pattern matching with support for '?' and '*'.

'?' Matches any single character.
'*' Matches any sequence of characters (including the empty sequence).

The matching should cover the entire input string (not partial).

The function prototype should be:
bool isMatch(const char *s, const char *p)

Some examples:
isMatch("aa","a") → false
isMatch("aa","aa") → true
isMatch("aaa","aa") → false
isMatch("aa", "*") → true
isMatch("aa", "a*") → true
isMatch("ab", "?*") → true
isMatch("aab", "c*a*b") → false

class Solution {
public:
    bool isMatch(const char *s, const char *p) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 递归解决
        // 第一次出错，runtime error ("", "*a*")
        // 第二次出错，timeout，因为重复*的出现
        // timeout...
        if ( p == NULL || s == NULL) {
            return false;
        }
       
        if (* p == 0) {
            return * s == 0;
        }
        if (* p != '*' && * p != '?') {
            if (* p == * s) { //当前字符匹配，进入下一个
                return isMatch( s + 1, p + 1);
            }
            else {  //当前字符不匹配，返回错误
                return false;
            }
        }
        else if (* p == '?') {  //匹配当前字符
            return isMatch( s + 1, p + 1);
        }
        else {
            if (* s == 0) {
                return isMatch( s, p + 1);
            }
            const char *end_p = p; //找到下一个不是'*'的位置
            while (*end_p && *end_p == '*') {
                end_p++;
            }
            return isMatch( s, end_p) || isMatch( s + 1, p); //匹配空字符或者匹配一个字符
        }
    }

    bool isMatch2( const char * s, const char * p) {
        if ( p == NULL || s == NULL) {
            return false;
        }
        // 88ms
        //思路就是记录p中最后一个*和s中当前位置，去匹配p*后面的和s的匹配情况，不匹配s++,然后重新循环
        int s_index = 0, p_index = 0;
        int s_len = strlen( s), p_len = strlen( p);
        int s_cur = 0, p_last_star = -1; //当前匹配的i  最后一个'*', -1表示没有匹配
        while (s_index < s_len) {
            //match 0 or multiple '*'
            while (p_index < p_len && p[p_index] == '*') {
                p_last_star = p_index++;
                s_cur = s_index;
            }
            if (p_index == p_len || ( p[p_index] != s[s_index] && p[p_index] != '?')) {
                if (p_last_star == -1) { //没有'*'，且p已经到头（或者p[p_index] != s[s_index] && p[p_index] != '?'）
                    return false;
                }
                else {      
                    s_index = ++s_cur; //s前进一个字符(因为当前s[s_cur] 只能和 * 去匹配，重新循环但是保证 ++s_cur
                    p_index = p_last_star + 1;
                }
            }
            else { //相等，进入下一位
                s_index++;
                p_index++;
            }
        }
        while (p_index < p_len && p[p_index] == '*') {
            p_index++;
        }
        return p_index == p_len;
    }

    bool isMatch3( const char * s, const char * p) {
        /*
          * c ? *
        a 1 0 0 0
        b 1 0 0 0
        c 1 1 0 0
        a 1 0 1 1
        b 1 0 0 1
        c 1 1 0 1

        如果当前patten 要匹配的字符是'*'，当前列为j 那么上一列为j - 1，上一列最早的1为lasttrue_position,则当前列从
        lasttrue_position开始到最后一行，都是true，因为*可以匹配任何字符，如果不是*，则只需要从lasttrue_position
        行开始找一直到最后一行，
        */
        // 统计p中非'*'的字符数
        int unchar_cnt = 0;
        const char *pt = p;
        while (*pt) {
            if (*pt != '*') {
                unchar_cnt++;
            }
            pt++;
        }
       
        string filter = "";
        int first = 0, cur = 0;
        int p_len = strlen( p);

        //不加入filter，348ms，加入filter 304ms
        for ( int i = 0; i < p_len;) {
            filter += p[i];
            if ( p[i] == '*') {
                while (i < p_len && p[i] == '*') {
                    i++;
                }
            }
            else {
                i++;
            }
        }
       
        p_len = filter.size();
        int s_len = strlen( s);
        if (unchar_cnt > s_len) { //肯定是错误的结果了
            return false;
        }
        vector< vector< bool>> dp(2, vector< bool>(s_len + 1));
       
        dp[0][0] = 1;
       
        for ( int j = 1; j <= p_len; j++) {
            //init zero
            dp[j % 2].assign(s_len + 1, 0);
            if (filter[j - 1] == '*') {
                for ( int i = 0; i <= s_len; i++) { //从最早的1开始
                    if (dp[(j - 1) % 2][i] == 1) {
                        for ( int k = i; k <= s_len; k++) {
                            dp[j % 2][k] = 1;
                        }
                        break;
                    }
                }
            }
            else {
                for ( int i = 1; i <= s_len; i++) {
                    dp[j % 2][i] = (filter[j - 1] == s[i - 1] || filter[j - 1] == '?') && dp[(j - 1) % 2][i - 1];
                }
            }
        }
        return dp[p_len % 2][s_len];
    }
};
