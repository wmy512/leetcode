Given a string s and a dictionary of words dict, add spaces in s to construct a sentence where each word is a valid dictionary word.

Return all such possible sentences.

For example, given
s = "catsanddog",
dict = ["cat", "cats", "and", "sand", "dog"].

A solution is ["cats and dog", "cat sand dog"].

#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <queue>
#include <stack>
#include <unordered_set>
#include <set>

class Solution {
public:
    vector<string> wordBreak( string s, unordered_set<string > &dict) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 先去判断有效无效
        if (wordBreakValid( s, dict) == false) {
            return vector< string>();
        }
        int min_len = INT_MAX, max_len = 0;
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            int word_len = (*iter).size();
            min_len = min(min_len, word_len);
            max_len = max(max_len, word_len);
        }
        vector< string> result;
        string one;
        dfs( s, dict, result, 0, one, min_len, max_len);
        return result;
    }
    void dfs(string &s, unordered_set< string> & dict, vector< string> & result, int index, string &one , int min_len , int max_len ) {
        if ( index == s.size()) {
            //cout << one << endl;
            result.push_back( one.substr(0, one.size() - 1));
        }
        for ( int i = index + min_len; i <= s.size() && i <= index + max_len; i++) {
            string word = s.substr( index, i - index);
            if ( dict.find(word) != dict.end()) {
                string origin = one;
                one += word + " ";
                dfs( s, dict, result, i, one, min_len, max_len);
                one = origin;
            }
        }
    }
     bool wordBreakValid( string s, unordered_set<string > &dict) {

        //真正的dp...类似背包的分配问题了
        vector< int> valid( s.size() + 1);
        valid[0] = 1;
        for ( int i = 0; i < s.size(); i++) {
            if (valid[i]) {
                for ( int j = 1; i + j <= s.size(); j++) {
                    if ( dict.find( s.substr(i, j)) != dict.end()) {
                        valid[i + j] = 1;
                    }
                }
            }
        }
        return valid[ s.size()];
    }

};
