Given a string s and a dictionary of words dict, determine if s can be segmented into a space-separated sequence of one or more dictionary words.

For example, given
s = "leetcode",
dict = ["leet", "code"].

Return true because "leetcode" can be segmented as "leet code".



#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <queue>
#include <stack>
#include <unordered_set>
#include <set>

//* Definition for singly-linked list.
struct ListNode {
    int val;
    ListNode *next;
    ListNode(int x) : val(x), next(NULL) {}
};

class Solution {
public:
    bool wordBreak( string s, unordered_set<string > &dict) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // timeout error
        // 加入预处理，如果s中含有字典中没有的字符直接退出
        // 可是是针对特例进行优化，不具有通用性，（12ms)
        int occur[256] = {0};
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            string word = *iter;
            for ( int i = 0; i < word.size(); i++) {
                if (occur[word[i]] == 0) {
                    occur[word[i]] = 1;
                }
            }
        }
        for ( int i = 0; i < s.size(); i++) {
            if (occur[ s[i]] == 0) {
                return false;
            }
        }

        return dfs( s, dict, 0);
    }
    bool dfs(string &s, unordered_set< string> & dict, int index) {
        if ( index == s.size()) {
            return true;
        }
        for ( int i = index + 1; i <= s.size(); i++) {
            string word = s.substr( index, i - index);
            if ( dict.find(word) != dict.end()) {
                if (dfs( s, dict, i)) {
                    return true;
                }
            }
        }
        return false;
    }
    bool wordBreak2( string s, unordered_set<string > &dict) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        // 28ms
        // 加入记忆功能，把后缀的匹配插入到一个set里面
       
        set< string> unmatched;
        return dfs2( s, dict, 0, unmatched);
    }
    bool dfs2(string &s, unordered_set< string> & dict, int index, set< string> &unmatched) {
        if ( index == s.size()) {
            return true;
        }
        for ( int i = index + 1; i <= s.size(); i++) {
            string word = s.substr( index, i - index);
            if ( dict.find(word) != dict.end()) { //必须首先进行这个的判断，否则suffix可能就添加了词典中的单词
                string suffix = s.substr(i);
                set< string>:: iterator unmatch_tmp = unmatched.find(suffix);  //这里是后缀，一开始写成word导致超时
                if (unmatch_tmp != unmatched.end()) { //不会匹配了
                    continue;
                }

                if (dfs2( s, dict, i, unmatched)) {
                    return true;
                }
                unmatched.insert(suffix);
            }
           
        }
        return false;
    }
    bool wordBreak3( string s, unordered_set<string > &dict) {

        //真正的dp...类似背包的分配问题了
        vector< int> valid( s.size() + 1);
        valid[0] = 1;
        for ( int i = 0; i < s.size(); i++) {
            if (valid[i]) {
                for ( int j = 1; i + j <= s.size(); j++) {
                    if ( dict.find( s.substr(i, j)) != dict.end()) {
                        valid[i + j] = 1;
                    }
                }
            }
        }
        return valid[ s.size()];
    }
};
