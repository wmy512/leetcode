Word Ladder

 Total Accepted: 1773 Total Submissions: 12029My Submissions

Given two words (start and end), and a dictionary, find the length of shortest transformation sequence from start to end, such that:

  1. Only one letter can be changed at a time
  2. Each intermediate word must exist in the dictionary

For example,

Given:
start = "hit"
end = "cog"
dict = ["hot","dot","dog","lot","log"]

As one shortest transformation is "hit" -> "hot" -> "dot" -> "dog" -> "cog",
return its length 5.

Note:

  • Return 0 if there is no such transformation sequence.
  • All words have the same length.
  • All words contain only lowercase alphabetic characters.

Discuss



class Solution {
public:
    int ladderLength( string start, string end, unordered_set< string> & dict) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        // 考虑到只有小写字母，所以直接从'a'到'z'找到不一样的替换即可产生下一个字符串，再用字典去重
//964ms
        if ( start == end) {
            return 1;
        }
        set< string> visited;
        queue< string> bfs;
        queue< int> step;
        bfs.push( start);
        step.push(1);

        visited.insert( start);

        while (!bfs.empty()) {
            int front_step = step.front();
            step.pop();
            string front_word = bfs.front();
            bfs.pop();
            for ( int i = 0; i < front_word.size(); i++) {
                for ( int j = 'a'; j <= 'z'; j++) {
                    if (j == front_word[i]) {
                        continue;
                    }
                    char tmp = front_word[i];
                    front_word[i] = j;
                    if (front_word == end) {
                        return front_step + 1;
                    }
                    if ( dict.find(front_word) != dict.end()) {
                        if (visited.find(front_word) == visited.end()) {
                            visited.insert(front_word);
                            bfs.push(front_word);
                            step.push(front_step + 1);
                        }
                    }
                    front_word[i] = tmp;
                }
            }
        }
       
           
        return 0;
    }
};
class Solution {
public:
    int ladderLength( string start, string end, unordered_set<string > &dict) {
        // IMPORTANT: Please reset any member data you declared, as
        // the same Solution instance will be reused for each test case.
        map< string, int> word_index; //改成unordered_map会将时间从1200ms降到1044ms
        int index = 0;
        dict.insert( start); //插入这两个元素
        dict.insert( end);
        //1200ms
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            word_index[*iter] = index++;
        }
        vector< vector< int> > path_map( dict.size());
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            string word = *iter;
            int index1 = word_index[word];
            for ( int i = 0; i < word.size(); i++) {
                for ( int j = 'a'; j <= 'z'; j++) {
                    if (word[i] == j) {
                        continue;
                    }
                    char tmp = word[i];
                    word[i] = j;
                    if ( dict.find(word) != dict.end()) {
                        int index2 = word_index[word];
                        path_map[index1].push_back(index2);
                    }
                    word[i] = tmp;
                }
            }
        }
        int start_index = word_index[ start];
        int end_index = word_index[ end];
        int step = FindPath(path_map, start_index, end_index);
        return step;

    }
    // bfs ...
    int FindPath(vector<vector<int> > & path_map, int start, int end) {
        queue< int> bfs;
        queue< int> step;
        bfs.push( start);
        step.push(1);
        vector< int> visited( path_map.size());
        visited[ start] = 1;
        while (!bfs.empty()) {
            int front = bfs.front();
            int front_step = step.front();
            step.pop();
            bfs.pop();
            for ( int i = 0; i < path_map[front].size(); i++) {
                int index = path_map[front][i];
                if (visited[index] == 0) {
                    visited[index] = 1;
                    bfs.push(index);
                    step.push(front_step + 1);
                    if (index == end) {
                        return front_step + 1;
                    }
                }
            }
        }
        return 0;

    }
};
