Given two words (start and end), and a dictionary, find all shortest transformation sequence(s) from start to end, such that:

  1. Only one letter can be changed at a time
  2. Each intermediate word must exist in the dictionary

For example,

Given:
start = "hit"
end = "cog"
dict = ["hot","dot","dog","lot","log"]

Return

  [
    ["hit","hot","dot","dog","cog"],
    ["hit","hot","lot","log","cog"]
  ]

Note:

  • All words have the same length.
  • All words contain only lowercase alphabetic characters.

    vector<vector<string>> ladderLength( string start, string end, unordered_set<string > &dict) {
        // IMPORTANT: Please reset any member data you declared, as

        dict.insert( start); //插入这两个元素
        dict.insert( end);
       
        map< string, vector< string> > path_map;
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            string word = *iter;
            vector< string> one_vec;
            path_map[word] = one_vec;
            string ori_word = word;
            for ( int i = 0; i < word.size(); i++) {
                for ( int j = 'a'; j <= 'z'; j++) {
                    if (word[i] == j) {
                        continue;
                    }
                    char tmp = word[i];
                    word[i] = j;
                    if ( dict.find(word) != dict.end()) {
                       
                        path_map[ori_word].push_back(word);
                    }
                    word[i] = tmp;
                }
            }
        }
      
        vector< vector< string> > result;
        FindPath(result, path_map, start, end);
        return result;

    }
    // bfs ...
    int FindPath(vector<vector<string>> & result, map< string, vector< string> > &path_map, string start , string end ) {
        queue< vector< string>> bfs;
        vector< string> one_level;
        one_level.push_back( start);
        bfs.push(one_level);

        set< string> visited;
        visited.insert( start);

        int step = 1;
        int min_step = INT_MAX;
        while (!bfs.empty()) {
            queue< vector< string>> bfs_tmp; //表示下一层的所有点
            while (!bfs.empty()) {
                vector< string> front = bfs.front(); //每一层最后一个节点都是需要访问的节点
                string first_word = *front.rbegin();
                bfs.pop();
               
                if (min_step != INT_MAX && step + 1 > min_step) {
                    break;
                }
                for ( int i = 0; i < path_map[first_word].size(); i++) {
                    string cur_word = path_map[first_word][i];
                    vector< string> one_chain(front);
                    one_chain.push_back(cur_word);
                    if (cur_word == end) {
                        result.push_back(one_chain);
                        min_step = step + 1;
                    }
                    else {
                        //插入新的路径
                        vector< string> one_chain(front);
                        one_chain.push_back(cur_word);
                        bfs_tmp.push(one_chain);
                    }
                }
            }
            swap(bfs, bfs_tmp); //使用swap技术，避免了添加一个关于索引的queue
            step++;
        }
        return 0;

    }
};


class Solution {
public:
    vector<vector<string>> findLadders( string start, string end, unordered_set<string > &dict) {
        //采用网上的另一种思路，保存两个vector,分别为上一次的节点，和本层要访问的节点
        //有个map<string, vector<string>>保存每个当前单词的前驱单词，这样一个递归就可以从end()访问所有路径，避免了所有的中间巨大内存消耗（也浪费了时间）
        unordered_map<string , vector <string >> prev_map;
        dict.insert( start);
        dict.insert( end);
        //初始化每个单词的前驱vector
        for ( unordered_set<string >::iterator iter = dict .begin(); iter != dict.end(); ++iter) {
            prev_map[*iter] = vector< string>();
        }

        vector< unordered_set<string >> candidate(2); //vector<string>会造成超时(为什么） // // unordered_set,就能通过，而且很快1420 ms
        int prev_index = 0;
        int cur_index = 1;
        candidate[prev_index].insert( start);

        while ( dict.size() > 0) {
            //delete prev vector element

            for ( unordered_set<string >::iterator iter = candidate[prev_index].begin(); iter != candidate[prev_index].end(); iter++) {
                dict.erase(*iter);
            }
            //clear cur vector
            candidate[cur_index].clear();
            //traversal
            int find_end = false;
            for ( unordered_set<string >::iterator iter = candidate[prev_index].begin(); iter != candidate[prev_index].end(); iter++) {
                string prev_word = *iter;
                for ( int j = 0; j < prev_word.size(); j++) {
                   
                    for ( int k = 'a'; k <= 'z'; k++) {
                        string cur_word = prev_word;
                        if (cur_word[j] == k) {
                            continue;
                        }
                        cur_word[j] = k;
                        if ( dict.find(cur_word) != dict.end()) {
                            candidate[cur_index].insert(cur_word);
                            prev_map[cur_word].push_back(prev_word);
                            if (cur_word == end) {
                                find_end = true; //用于提前终止循环
                            }
                                
                        }
                    }
                }
            }
            if (find_end == true) {
                break;
            }
            if (candidate[cur_index].size() == 0) {
                return vector< vector< string>>();
            }
           
            int tmp = cur_index;
            cur_index = prev_index;
            prev_index = tmp;
        }
        vector< vector< string>> result;
        vector< string> path;
        BuildPath(result, prev_map, path, end);
        return result;

    }
    void BuildPath( vector< vector< string>> & result, unordered_map<string , vector<string >> &prev_map, vector<string> & path, string word) {
        if ( prev_map[ word].size() == 0) { //递归终止条件，到达最开始的单词
            //path.push_back(word);
            vector< string> one_solution( path);
            one_solution.push_back( word);
            reverse(one_solution.begin(), one_solution.end());
            result.push_back(one_solution);
            return;
        }
        path.push_back( word);
        for ( int i = 0; i < prev_map[ word].size(); i++) {
            BuildPath( result, prev_map, path, prev_map[ word][i]);
        }
        path.pop_back();
    }
};
