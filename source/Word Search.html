Given a 2D board and a word, find if the word exists in the grid.

The word can be constructed from letters of sequentially adjacent cell, where "adjacent" cells are those horizontally or vertically neighboring. The same letter cell may not be used more than once.

For example,
Given board =

[
  ["ABCE"],
  ["SFCS"],
  ["ADEE"]
]
word = "ABCCED", -> returns true,
word = "SEE", -> returns true,
word = "ABCB", -> returns false.

Discuss


#include <iostream>
using namespace std;
#include <vector>
#include <string>
#include <algorithm>


class Solution {
public:
    bool exist(vector<vector<char> > & board, string word) {
        // Note: The Solution object is instantiated only once and is reused by each test case.
        if ( word.size() <= 0) {
            return true;
        }
        if ( board.size() == 0) {
            return false;
        }
        vector< vector< int> > visited( board.size(), vector< int>( board[0].size()));
        bool flag = false;
        //第一次出错：原因，应该是先找好起始点，而且然后再去判断，其它方向
        //第二次出错：超时(board的ij没有写好。。。
        //第三次因为初始访问的那个元素没有标记为已经访问
        for ( int i = 0; i < board.size(); i++) {
            for ( int j = 0; j < board[i].size(); j++) {
                if ( board[i][j] == word[0]) {
                    visited.assign( board.size(), vector< int>( board[0].size()));
                    visited[i][j] = 1;
                    dfs( board, word, visited, i, j, 1, flag);   //这是ij
                    if (flag == true) {
                        return flag;
                    }
                }
            }
        }
        return flag;
    }
    void dfs(vector<vector<char> > & board, string word, vector< vector< int> > &visited, int x , int y , int index , bool &flag ) {
        if ( index == word.size()) {
            flag = true;
            return;
        }
        if ( flag == false) {
            for ( int i = 0; i < 4; i++) {
                int x_new = x + x_pos[i];
                int y_new = y + y_pos[i];
                if (x_new >= 0 && x_new < board.size() && y_new >= 0 && y_new < board[0].size()
                    && ! visited[x_new][y_new] && ! flag && board[x_new][y_new] == word[index ]) {
                    visited[x_new][y_new] = 1;
                    dfs( board, word, visited, x_new, y_new, index + 1, flag);
                    visited[x_new][y_new] = 0;
                }
            }
        }
    }
    Solution() {
        int y_tmp[4] = {1, 0, -1, 0};  // right, down, left, up(垂直x方向)
        int x_tmp[4] = {0, 1, 0, -1};
        for ( int i = 0; i < 4; i++) {
            x_pos[i] = x_tmp[i];
            y_pos[i] = y_tmp[i];
        }
    }
private:
    int x_pos[4];
    int y_pos[4];

};

int main() {

    for (int i = 0; i < 1; i++) {

    /*for (int i = 0; i < 3; i++) {

}
